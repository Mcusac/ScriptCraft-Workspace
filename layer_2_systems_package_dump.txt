package com.weylandyutani.layer_2_systems.level_0.ai;

import com.weylandyutani.layer_0_domain.level_0.spatial.BlockPosRef;
import com.weylandyutani.layer_0_domain.level_0.util.BoundedIntRandom;
import java.util.function.Predicate;

/**
 * Expanding-radius block position search (algorithm only; probes supplied by L3).
 */
public final class BlockPosSearchPolicy {

    private BlockPosSearchPolicy() {}

    public static BlockPosRef searchInRadius(
            BlockPosRef center,
            double maxRadius,
            int attemptsPerRadius,
            BoundedIntRandom random,
            Predicate<BlockPosRef> predicate) {
        if (center == null || random == null || predicate == null) {
            return null;
        }

        for (int radius = 5; radius <= maxRadius; radius += 5) {
            for (int i = 0; i < attemptsPerRadius; i++) {
                BlockPosRef candidate = center.offset(
                        random.nextInt(radius * 2) - radius,
                        random.nextInt(4) - 2,
                        random.nextInt(radius * 2) - radius);
                if (predicate.test(candidate)) {
                    return candidate;
                }
            }
        }
        return null;
    }
}

/**
 * Resolves AI goal slot priorities from entity JSON via {@link EntityConfigStorage}.
 * No Java fallbacks — missing keys throw {@link ConfigContractException}.
 */
public final class GoalPriorityResolver {

    private static final String RESOLVER = "GoalPriorityResolver";

    private GoalPriorityResolver() {
        throw new UnsupportedOperationException("GoalPriorityResolver is a utility class and cannot be instantiated");
    }

    /**
     * Returns the configured priority for a goal key on the given xenomorph type.
     *
     * @param type    species owning the priority map
     * @param goalKey key in {@code ai_goals.goal_priorities}
     * @return priority integer (lower = higher precedence in Minecraft goal selector)
     */
    public static int priority(XenomorphType type, String goalKey) {
        EntityConfig config = EntityConfigStorage.getConfig(type);
        if (config.aiGoals == null) {
            throw new ConfigContractException(RESOLVER, type.name(), "ai_goals", "required");
        }
        if (config.aiGoals.goalPriorities == null) {
            throw new ConfigContractException(RESOLVER, type.name(), "ai_goals.goal_priorities", "required");
        }
        Integer priority = config.aiGoals.goalPriorities.get(goalKey);
        if (priority == null) {
            throw new ConfigContractException(
                    RESOLVER, type.name(), "ai_goals.goal_priorities." + goalKey, "required");
        }
        return priority;
    }
}

/**
 * Pure suffocation decision policy for xenomorph survival AI.
 * Thresholds are ratios of current air supply to max air supply (0.0–1.0).
 */
public final class SurvivalSuffocationPolicy {

    private SurvivalSuffocationPolicy() {
        throw new UnsupportedOperationException("SurvivalSuffocationPolicy is a utility class");
    }

    public static boolean isSuffocating(
            boolean inWall,
            boolean stuck,
            int airSupply,
            int maxAirSupply,
            double lowThreshold,
            double criticalThreshold) {
        if (inWall) {
            return true;
        }
        if (maxAirSupply <= 0) {
            return false;
        }

        double airRatio = (double) airSupply / maxAirSupply;
        if (airRatio <= criticalThreshold) {
            return true;
        }
        return stuck && airRatio <= lowThreshold;
    }
}

package com.weylandyutani.layer_2_systems.level_0.ai;

import com.weylandyutani.layer_0_domain.level_0.spatial.Vec3Ref;
import com.weylandyutani.layer_0_domain.level_0.util.BoundedIntRandom;
import java.util.function.Predicate;

/**
 * Directional Vec3 position search (algorithm only; validity probes supplied by L3).
 */
public final class Vec3SearchPolicy {

    private Vec3SearchPolicy() {}

    public static Vec3Ref search(
            Vec3Ref origin,
            Vec3Ref baseDirection,
            double baseDistance,
            int maxAttempts,
            BoundedIntRandom random,
            Predicate<Vec3Ref> predicate) {
        if (origin == null || baseDirection == null || random == null || predicate == null) {
            return null;
        }

        Vec3Ref direction = baseDirection.normalize();
        for (int i = 0; i < maxAttempts; i++) {
            Vec3Ref candidate = origin.add(direction.scale(baseDistance + random.nextInt(8)));
            if (predicate.test(candidate)) {
                return candidate;
            }
            direction = direction.yRot((float) (Math.PI / 4));
        }
        return null;
    }
}

package com.weylandyutani.layer_2_systems.level_0.combat;

import com.weylandyutani.layer_0_domain.level_0.spatial.BlockPosRef;
import com.weylandyutani.layer_0_domain.level_0.spatial.WorldRef;
import com.weylandyutani.layer_0_domain.level_1.world.WorldMutationPort;
import com.weylandyutani.layer_1_infrastructure.level_0.platform.ModLogger;
import com.weylandyutani.layer_1_infrastructure.level_1.block.BlockRulesStorage;

import java.util.HashSet;
import java.util.Set;

/**
 * Policy-only block damage orchestration for xenomorphs.
 * 
 * REFACTORED: Now uses {@link BlockRulesStorage} instead of domain-layer UnbreakableBlockIds.
 * Unbreakable blocks are loaded from config (block_rules.json).
 * Breakable blocks can be whitelisted via config or pattern-matched (glass, door, fence, wall).
 */
public class BlockDamageOrchestrator {

    private static boolean isEnabled = false;
    private static final Set<String> breakableBlocks = new HashSet<>();

    private BlockDamageOrchestrator() {
    }

    public static void initialize() {
        try {
            isEnabled = true;
            if (isEnabled) {
                loadBreakableBlocks();
                ModLogger.info("BlockDamageOrchestrator", "🔨 Block Damage System enabled");
            } else {
                ModLogger.info("BlockDamageOrchestrator", "🔒 Block Damage System disabled");
            }
        } catch (Exception e) {
            ModLogger.warn("BlockDamageOrchestrator", "⚠️ Failed to initialize Block Damage System: " + e.getMessage());
            isEnabled = false;
        }
    }

    public static boolean isEnabled() {
        return isEnabled;
    }

    /**
     * Attempt to damage a block when policy allows.
     *
     * @param mutation domain mutation port for world operations
     * @param world domain world reference
     * @param pos block position
     * @param blockId registry id such as {@code minecraft:glass}
     * @param actorLabel logging label for the breaking entity
     * @return true when the block was broken
     */
    public static boolean attemptBlockDamage(
            WorldMutationPort mutation, WorldRef world, BlockPosRef pos, String blockId, String actorLabel) {
        if (!isEnabled || mutation == null || world == null || pos == null || blockId == null || blockId.isBlank()) {
            return false;
        }
        
        // First check: is this block type breakable at all?
        if (!canBreakBlock(blockId)) {
            return false;
        }
        
        // Second check: is this specific block unbreakable per configuration?
        if (isUnbreakableBlock(blockId)) {
            return false;
        }

        boolean broken = mutation.breakBlock(world, pos);
        if (broken) {
            ModLogger.debug("BlockDamageOrchestrator",
                    "🔨 " + actorLabel + " broke block " + blockId + " at " + pos);
        }
        return broken;
    }

    /**
     * Check if a block type can be broken.
     * Checks:
     * 1. Config whitelist (explicit breakable blocks)
     * 2. Hardcoded list fallback
     * 3. Pattern matching (glass, door, fence, wall)
     *
     * @param blockId registry ID
     * @return true if this block type can be broken
     */
    private static boolean canBreakBlock(String blockId) {
        // Check if explicitly in config breakable whitelist
        if (BlockRulesStorage.isLoaded() && BlockRulesStorage.getConfig().isExplicitlyBreakable(blockId)) {
            return true;
        }
        
        // Fallback to hardcoded set
        if (breakableBlocks.contains(blockId)) {
            return true;
        }
        
        // Pattern matching fallback
        return blockId.contains("glass")
                || blockId.contains("door")
                || blockId.contains("fence")
                || blockId.contains("wall");
    }

    /**
     * Check if a block is unbreakable per configuration.
     * Delegates to {@link BlockRulesStorage}.
     *
     * @param blockId registry ID
     * @return true if this block cannot be broken
     */
    private static boolean isUnbreakableBlock(String blockId) {
        if (!BlockRulesStorage.isLoaded()) {
            ModLogger.warn("BlockDamageOrchestrator", "BlockRules config not loaded; treating as breakable");
            return false;
        }
        return BlockRulesStorage.getConfig().isUnbreakable(blockId);
    }

    /**
     * Load the fallback pattern-based breakable blocks list.
     * This is used when blocks are not in the explicit config whitelist.
     * Examples: "minecraft:glass", "minecraft:oak_door", etc.
     */
    private static void loadBreakableBlocks() {
        breakableBlocks.add("minecraft:glass");
        breakableBlocks.add("minecraft:white_stained_glass");
        breakableBlocks.add("minecraft:black_stained_glass");
        breakableBlocks.add("minecraft:oak_door");
        breakableBlocks.add("minecraft:iron_door");
        breakableBlocks.add("minecraft:oak_fence");
        breakableBlocks.add("minecraft:glass_pane");
    }

    public static void reload() {
        initialize();
    }
}

package com.weylandyutani.layer_2_systems.level_0.growth;

import com.weylandyutani.layer_0_domain.level_0.entity.EntityThreatLevel;
import com.weylandyutani.layer_0_domain.level_1.entity.XenomorphType;
import com.weylandyutani.layer_1_infrastructure.level_3.lifecycle.LifecycleConfig;

/**
 * Resolver for xenomorph entity form transitions based on lifecycle stage.
 * Maps entity lifecycle data to appropriate xenomorph form (drone, queen, etc.).
 *
 * <p><strong>Authority:</strong> {@code lifecycle.json} {@code form_thresholds} via {@link LifecycleConfig}.
 *
 * <p><strong>Runtime note:</strong> Live entity form identity uses {@code XenomorphFormQueryPort}
 * (NeoForge adapter + entity type). This resolver is pre-wired infrastructure for future
 * maturation/threat-driven form promotion — no production caller yet (MIG-DEF-005 closed).
 *
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class EntityFormResolver {
    
    /**
     * Resolve xenomorph form based on lifecycle parameters.
     * Determines which form (drone, queen, praetor, etc.) the entity should take.
     * 
     * @param baseType Base xenomorph type from host
     * @param maturationStage Current maturation stage (0-100)
     * @param threatLevel Current threat level assessment
     * @return Resolved xenomorph form
     */
    public XenomorphType resolveForm(XenomorphType baseType, int maturationStage, EntityThreatLevel threatLevel) {
        if (baseType == null) {
            return XenomorphType.DRONE;
        }
        
        if (maturationStage < LifecycleConfig.getEarlyStageExclusiveThreshold()) {
            return XenomorphType.DRONE;
        }
        
        if (maturationStage < LifecycleConfig.getMidStageExclusiveThreshold()) {
            return XenomorphType.DRONE;
        }
        
        if (maturationStage < LifecycleConfig.getLateStageExclusiveThreshold()) {
            if (threatLevel == EntityThreatLevel.HIGH || threatLevel == EntityThreatLevel.CRITICAL) {
                return XenomorphType.PRAETOR;
            }
            return XenomorphType.DRONE;
        }
        
        if (maturationStage >= LifecycleConfig.getQueenMaturationMinThreshold()
                && threatLevel == EntityThreatLevel.CRITICAL) {
            return XenomorphType.QUEEN;
        }
        
        return XenomorphType.PRAETOR;
    }
    
    /**
     * Check if form transition is allowed at current stage.
     * 
     * @param currentForm Current form
     * @param targetForm Target form to transition to
     * @param maturationStage Current maturation (0-100)
     * @return true if transition is allowed
     */
    public boolean canTransition(XenomorphType currentForm, XenomorphType targetForm, int maturationStage) {
        if (currentForm == targetForm) {
            return true;
        }
        
        if (currentForm == XenomorphType.DRONE
                && targetForm == XenomorphType.PRAETOR
                && maturationStage >= LifecycleConfig.getDroneToPraetorMinThreshold()) {
            return true;
        }
        
        if (currentForm == XenomorphType.PRAETOR
                && targetForm == XenomorphType.QUEEN
                && maturationStage >= LifecycleConfig.getQueenMaturationMinThreshold()) {
            return true;
        }
        
        return false;
    }
}

package com.weylandyutani.layer_2_systems.level_0.growth;

import com.weylandyutani.layer_0_domain.level_0.validation.ValidationResult;

/**
 * Territory expansion validation logic for hive growth.
 * Validates whether a queen can expand territory to a new region.
 * 
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class TerritoryExpansionValidator {
    
    private final int minDistanceBetweenNests;
    private final int maxTerritoryRadius;
    private final float minClearancePercentage;
    
    public TerritoryExpansionValidator(int minDistanceBetweenNests, int maxTerritoryRadius,
                                     float minClearancePercentage) {
        this.minDistanceBetweenNests = minDistanceBetweenNests;
        this.maxTerritoryRadius = maxTerritoryRadius;
        this.minClearancePercentage = minClearancePercentage;
    }
    
    /**
     * Validate if territory can be expanded at given location.
     * Checks clearance, proximity to existing nests, and expansion limits.
     * 
     * @param currentCenterX Current nest center X
     * @param currentCenterZ Current nest center Z
     * @param expansionX Proposed expansion X
     * @param expansionZ Proposed expansion Z
     * @param clearancePercentage Percentage of clear space (0.0-1.0)
     * @return Validation result with success/failure reasons
     */
    public ValidationResult validateExpansion(int currentCenterX, int currentCenterZ,
                                            int expansionX, int expansionZ,
                                            float clearancePercentage) {
        // Check clearance threshold
        if (clearancePercentage < minClearancePercentage) {
            return ValidationResult.failure(
                String.format("Insufficient clearance: %.1f%% available, %.1f%% required",
                    clearancePercentage * 100, minClearancePercentage * 100));
        }
        
        // Check territory radius limit
        double distance = Math.sqrt(
            Math.pow(expansionX - currentCenterX, 2) + 
            Math.pow(expansionZ - currentCenterZ, 2)
        );
        if (distance > maxTerritoryRadius) {
            return ValidationResult.failure(
                String.format("Expansion too far: %.1f blocks, max %.0f blocks", 
                    distance, (float) maxTerritoryRadius));
        }
        
        return ValidationResult.success();
    }
    
    public int getMinDistanceBetweenNests() {
        return minDistanceBetweenNests;
    }
    
    public int getMaxTerritoryRadius() {
        return maxTerritoryRadius;
    }
    
    public float getMinClearancePercentage() {
        return minClearancePercentage;
    }
}

package com.weylandyutani.layer_2_systems.level_0.hive;

import java.util.function.Predicate;

/**
 * Orchestration workflow for egg laying behavior.
 * Encapsulates timing, location search, and policy enforcement.
 *
 * Static utility class — no instance state.
 * Each AI behavior maintains its own timer and calls these stateless methods.
 *
 * Boundaries:
 * - Does NOT import Minecraft types
 * - Does NOT perform block placement
 * - Does NOT own timing state
 */
public final class EggLayingWorkflow {

    private EggLayingWorkflow() {
    }

    /**
     * Advance a timer by one tick.
     *
     * @param currentTimer current timer value
     * @param interval threshold for reset
     * @return TickResult with next timer value and whether threshold was reached
     */
    public static TickResult advanceTimer(int currentTimer, int interval) {
        int next = currentTimer + 1;
        return new TickResult(next >= interval ? 0 : next, next >= interval);
    }

    /**
     * Find a suitable candidate location for egg placement within limits.
     * Searches in expanding rings from center, testing with provided predicate.
     *
     * @param center search center position
     * @param limits search area and density limits
     * @param suitability test for location viability
     * @return candidate position or null if none found
     */
    public static BlockPosRef findCandidateLocation(BlockPosRef center, Limits limits, Predicate<BlockPosRef> suitability) {
        if (center == null || limits == null || suitability == null) {
            return null;
        }
        int radius = limits.eggPlacementRadius();
        for (int ring = 0; ring <= radius; ++ring) {
            for (int x = -ring; x <= ring; ++x) {
                for (int z = -ring; z <= ring; ++z) {
                    if (Math.abs(x) != ring && Math.abs(z) != ring) {
                        continue;
                    }
                    for (int y = -limits.verticalSearchRange(); y <= limits.verticalSearchRange(); ++y) {
                        BlockPosRef candidate = new BlockPosRef(center.x() + x, center.y() + y, center.z() + z);
                        if (!suitability.test(candidate)) {
                            continue;
                        }
                        return candidate;
                    }
                }
            }
        }
        return null;
    }

    /**
     * Count eggs present in search area.
     *
     * @param world world reference
     * @param center search center
     * @param limits search area limits
     * @return egg count
     */
    public static int countEggsInArea(WorldRef world, BlockPosRef center, Limits limits) {
        return 0;
    }

    /**
     * Check if egg laying is allowed based on density limits.
     *
     * @param eggCount count in area
     * @param chunkEggCount count in chunk
     * @param limits policy limits
     * @return true if laying is permitted
     */
    public static boolean canLayInArea(int eggCount, int chunkEggCount, Limits limits) {
        if (limits == null) {
            return false;
        }
        return EggLayingPolicy.canLayEgg(eggCount, limits.maxEggsInArea())
                && EggLayingPolicy.isChunkBelowEggLimit(chunkEggCount, limits.maxEggsPerChunk());
    }

    /**
     * Check if location is clear of nearby eggs.
     *
     * @param world world reference
     * @param pos location to check
     * @param minDistance minimum distance to other eggs
     * @return true if location is clear
     */
    public static boolean isLocationClearOfEggs(WorldRef world, BlockPosRef pos, int minDistance) {
        return pos != null;
    }

    /**
     * Result of a timer tick operation.
     */
    public record TickResult(int nextTimer, boolean shouldAttemptLay) {
    }

    /**
     * Search and density limits for egg laying.
     */
    public record Limits(
            int searchRadius,
            int maxEggsPerChunk,
            int maxEggsInArea,
            int eggPlacementRadius,
            int minDistanceBetweenEggs,

    }

    public static int getHiveStructureSize() {
        return 9;
    }

    public static int getTotalPhases() {
        return 5;
    }

    public static List<PlacementIntent> planPhase(int phase, int structureSize, Random random) {
        List<PlacementIntent> intents = new ArrayList<>();
        int radius = Math.max(2, structureSize / 2);
        switch (phase) {
            case 0 -> {
                for (int x = -radius; x <= radius; x++) {
                    for (int z = -radius; z <= radius; z++) {
                        intents.add(new PlacementIntent(new BlockOffset(x, 0, z), BlockKind.RESIN));
                    }
                }
            }
            case 1 -> {
                for (int x = -radius; x <= radius; x++) {
                    for (int z = -radius; z <= radius; z++) {
                        if (Math.abs(x) == radius || Math.abs(z) == radius) {
                            intents.add(new PlacementIntent(new BlockOffset(x, 1, z), BlockKind.RESIN));
                        }
                    }
                }
            }
            case 2 -> intents.add(new PlacementIntent(new BlockOffset(0, 1, 0), BlockKind.SPINY));
            case 3 -> intents.add(new PlacementIntent(new BlockOffset(0, 0, 0), BlockKind.AIR_ENTRANCE));
            case 4 -> intents.add(new PlacementIntent(new BlockOffset(1, 1, 1), BlockKind.COCOON));
            default -> {
            }
        }
        return intents;
    }

package com.weylandyutani.layer_2_systems.level_0.ovomorph;

/**
 * Orchestration workflow for ovomorph hatching lifecycle.
 * Encapsulates timing, proximity requirements, and hatch trigger logic.
 *
 * Static utility class with mutable state record.
 * Each ovomorph entity maintains its own State instance.
 *
 * Boundaries:
 * - Does NOT import Minecraft types
 * - Does NOT perform entity spawning (delegates to port)
 * - Does NOT validate hosts (delegates to port)
 * - Manages hatching timer and state progression
 */
public final class OvomorphHatchWorkflow {

    private OvomorphHatchWorkflow() {
    }

    /**
     * Initialize hatching state when hatching begins.
     *
     * @param port the hatching port interface
     * @param state the state to initialize
     */
    public static void onHatchingStarted(OvomorphHatchPort port, State state) {
        state.startTick = -1;
        state.elapsedTicks = 0;
    }

    /**
     * Process one tick of the hatching lifecycle.
     * Checks proximity requirements, advances timer, and triggers hatch when ready.
     *
     * @param port the hatching port interface
     * @param state the current hatching state
     * @param currentTick the server tick count
     */
    public static void processTick(OvomorphHatchPort port, State state, int currentTick) {
        // Initialize start tick on first call
        if (state.startTick < 0) {
            state.startTick = currentTick;
        }

        // Calculate elapsed time
        state.elapsedTicks = currentTick - state.startTick;

        // Cancel if proximity required but no host nearby
        if (port.isProximityRequired() && !port.isViableHostNearby()) {
            port.cancelHatching();
            return;
        }

        // Update state
        state.hatchTimer = state.elapsedTicks;
        state.wasHostNearby = port.isViableHostNearby();

        // Execute hatch if time threshold reached
        if (state.elapsedTicks >= port.hatchTime()) {
            port.executeHatch();
        }
    }

    /**
     * Mutable state for ovomorph hatching process.
     * Each ovomorph instance maintains one State throughout its lifetime.
     */
    public static final class State {
        public int hatchTimer;
        public boolean wasHostNearby;
        private int startTick = -1;
        private int elapsedTicks;
    }
}

            boolean hostTargeted,
            boolean hostReserved,

        return hostTargeted || hostReserved || hostAttached;
    }

        return validTarget;
    }

    /**
     * Another facehugger owns {@code HOST_ATTACHED} for this host (this one should stop).
     */
    public static boolean isHostAttachedByOther(
            boolean hasHostAttachedSignal,
            boolean thisFacehuggerOwnsActiveSignal) {
        if (!hasHostAttachedSignal) {
            return false;
        }
        return !thisFacehuggerOwnsActiveSignal;
    }

    public static boolean shouldContinueAttachment(
            boolean hostInvalid,
            boolean hostAttachedByOther,
            boolean attachmentInProgress) {
        if (hostInvalid || hostAttachedByOther) {
            return false;
        }
        return attachmentInProgress;
    }

    public static double resolveFollowRange(double configFollowRange, double attributeFollowRange) {
        if (configFollowRange > 0) {
            return configFollowRange;
        }
        if (attributeFollowRange > 0) {
            return attributeFollowRange;
        }
        throw new IllegalStateException("Facehugger follow_range missing from entity JSON and entity attributes");
    }

    /**
     * Approach-tick decision for the attachment goal (move vs stop + FSM state).
     */
    public record ApproachTickDecision(
            AttachmentState nextState,
            boolean shouldMoveTowardHost,
            boolean shouldStopNavigation) {
    }

    public static ApproachTickDecision resolveApproachTick(
            AttachmentState currentState,
            double distanceToTarget,
            double minDistance) {
        AttachmentState nextState = AttachmentRules.nextApproachState(currentState, distanceToTarget, minDistance);
        if (distanceToTarget > minDistance) {
            return new ApproachTickDecision(nextState, true, false);
        }
        boolean stateChanged = nextState != currentState;
        return new ApproachTickDecision(nextState, false, stateChanged);
    }

        return state == AttachmentState.ATTACHING || state == AttachmentState.ATTACHED;
    }

    public static boolean shouldInitializeRotationLock(AttachmentState state, boolean targetAlive, boolean rotationAlreadySet) {
        return shouldLockRotation(state) && targetAlive && !rotationAlreadySet;
    }

    public static boolean shouldClearRotationLock(AttachmentState state) {
        return !shouldLockRotation(state);
    }

    public static float calculateYawToFaceTarget(
            double facehuggerX,
            double facehuggerZ,
            double targetX,
            double targetZ,
            float currentYaw) {
        double dx = targetX - facehuggerX;
        double dz = targetZ - facehuggerZ;
        double distance = Math.sqrt(dx * dx + dz * dz);
        if (distance < 0.1) {
            return currentYaw;
        }
        return (float) (Math.atan2(dz, dx) * 180.0 / Math.PI) - 90.0f;
    }
}

    public static void loadSystems() {
        ModLogger.info("🔧 Loading modular systems from " + REGISTRIES.size() + " registries...");

        int loadedCount = 0;
        int failedCount = 0;

        for (SystemRegistry registry : REGISTRIES) {
            if (registry.isEnabled()) {
                try {
                    ModLogger.debug("Initializing system: " + registry.getSystemName());
                    registry.initialize();
                    loadedCount++;
                } catch (Exception e) {
                    ModLogger.error("Failed to initialize system " + registry.getSystemName() + ": " + e.getMessage());
                    failedCount++;
                }
            } else {
                ModLogger.debug("Skipping disabled system: " + registry.getSystemName());
            }
        }

        if (failedCount > 0) {
            ModLogger.warn("⚠️ System loading completed with " + loadedCount + " successful and " + failedCount + " failed");
        } else {
            ModLogger.info("✅ All systems loaded successfully (" + loadedCount + " systems)");
        }
    }

    public static void reloadSystems() {
        ModLogger.info("🔄 Reloading systems...");

        int reloadedCount = 0;
        int failedCount = 0;

        for (SystemRegistry registry : REGISTRIES) {
            if (registry.isEnabled()) {

                    registry.reload();
                    reloadedCount++;
                } catch (Exception e) {
                    ModLogger.error("Failed to reload system " + registry.getSystemName() + ": " + e.getMessage());

    private final BlockPosRef position;
    private final double radius;
    private final long creationTick;
    private final int durationTicks;
    private final Map<String, Object> metadata;

    public AISignalData(
            AISignalType type,
            UUID sourceEntityId,
            BlockPosRef position,
            double radius,
            long creationTick,
            int durationTicks) {
        this.type = type;
        this.sourceEntityId = sourceEntityId;
        this.position = position;
        this.radius = radius;
        this.creationTick = creationTick;
        this.durationTicks = durationTicks;
        this.metadata = new HashMap<>();
    }

    public AISignalType getType() {
        return type;
    }

    public UUID getSourceEntityId() {
        return sourceEntityId;
    }

    public BlockPosRef getPosition() {
        return position;
    }

    public double getRadius() {
        return radius;
    }

    public long getCreationTick() {
        return creationTick;
    }

    public int getDurationTicks() {
        return durationTicks;
    }

    public Map<String, Object> getMetadata() {
        return metadata;
    }

    public boolean isExpired(long currentTick) {
        return (currentTick - creationTick) > durationTicks;
    }

    public double getIntensity(BlockPosRef checkPos) {
        if (checkPos == null) {
            return 0.0;
        }
        double distance = Math.sqrt(position.distanceSquared(checkPos));
        if (distance > radius) {
            return 0.0;
        }
        return 1.0 - (distance / radius);
    }

package com.weylandyutani.layer_2_systems.level_0.signals;

import com.weylandyutani.layer_0_domain.level_0.ai.AISignalType;
import com.weylandyutani.layer_0_domain.level_0.spatial.BlockPosRef;
import com.weylandyutani.layer_0_domain.level_1.entity.LivingEntityRef;
import com.weylandyutani.layer_1_infrastructure.level_0.platform.ModLogger;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import java.util.UUID;
import java.util.concurrent.ConcurrentHashMap;

/**
 * Hybrid AI signal coordination store (domain protocol + world position binding).
 */
public class AISignalSystem {

    private static final Map<AISignalType, List<AISignalData>> activeSignals = new ConcurrentHashMap<>();
    private static final Map<UUID, List<AISignalData>> entitySignals = new ConcurrentHashMap<>();

    private AISignalSystem() {}

    public static AISignalData broadcastSignal(
            AISignalType type,
            LivingEntityRef sourceEntity,
            BlockPosRef position,
            double radius,
            long gameTime,
            int durationTicks,
            String sourceLabel) {
        if (sourceEntity == null || position == null) {
            ModLogger.warn("AISignalSystem", "⚠️ Cannot broadcast signal - source or position is null");
            return null;
        }

        AISignalData signal = new AISignalData(
                type, sourceEntity.entityId(), position, radius, gameTime, durationTicks);
        activeSignals.computeIfAbsent(type, k -> new ArrayList<>()).add(signal);
        entitySignals.computeIfAbsent(sourceEntity.entityId(), k -> new ArrayList<>()).add(signal);

        ModLogger.debug(
                "AISignalSystem",
                "📡 Broadcast signal: " + type + " from " + sourceLabel + " at "
                        + position + " (radius: " + radius + ", duration: " + durationTicks + " ticks)");

        return signal;
    }

    public static AISignalData getSignal(AISignalType type, BlockPosRef position, double maxDistance, long currentTick) {
        List<AISignalData> signals = activeSignals.get(type);
        if (signals == null || signals.isEmpty()) {
            return null;
        }

        AISignalData strongest = null;
        double strongestIntensity = 0.0;

        for (AISignalData signal : signals) {
            if (signal.isExpired(currentTick)) {
                continue;
            }
            double intensity = signal.getIntensity(position);
            if (intensity > strongestIntensity && intensity > 0.0) {
                strongest = signal;
                strongestIntensity = intensity;
            }
        }
        return strongest;
    }

    public static boolean hasSignalForEntity(AISignalType type, UUID entityId, long currentTick) {
        List<AISignalData> signals = activeSignals.get(type);
        if (signals == null || signals.isEmpty()) {
            return false;
        }

        for (AISignalData signal : signals) {
            if (signal.isExpired(currentTick)) {
                continue;
            }
            Object targetId = signal.getMetadata().get("targetEntityId");
            if (targetId instanceof UUID && targetId.equals(entityId)) {
                return true;
            }
        }
        return false;
    }

    public static boolean hasSignalForEntityExcludingSource(
            AISignalType type, UUID entityId, UUID excludeSourceId, long currentTick) {
        List<AISignalData> signals = activeSignals.get(type);
        if (signals == null || signals.isEmpty()) {
            return false;
        }

        for (AISignalData signal : signals) {
            if (signal.isExpired(currentTick)) {
                continue;
            }
            if (excludeSourceId != null && signal.getSourceEntityId().equals(excludeSourceId)) {
                continue;
            }
            Object targetId = signal.getMetadata().get("targetEntityId");
            if (targetId instanceof UUID && targetId.equals(entityId)) {
                return true;
            }
        }
        return false;
    }

    public static List<AISignalData> getSignalsNear(BlockPosRef position, double maxDistance, long currentTick) {
        List<AISignalData> results = new ArrayList<>();

        for (List<AISignalData> signals : activeSignals.values()) {
            for (AISignalData signal : signals) {
                if (signal.isExpired(currentTick)) {
                    continue;
                }
                double intensity = signal.getIntensity(position);
                if (intensity > 0.0) {
                    results.add(signal);
                }
            }
        }

        results.sort((a, b) -> Double.compare(b.getIntensity(position), a.getIntensity(position)));
        return results;
    }

    public static void cleanupExpiredSignals(long currentTick) {
        for (List<AISignalData> signals : activeSignals.values()) {
            signals.removeIf(signal -> signal.isExpired(currentTick));
        }
        for (List<AISignalData> signals : entitySignals.values()) {
            signals.removeIf(signal -> signal.isExpired(currentTick));
        }
    }

    public static void clearSignalsForEntity(UUID entityId) {
        entitySignals.remove(entityId);
        for (List<AISignalData> signals : activeSignals.values()) {
            signals.removeIf(signal -> signal.getSourceEntityId().equals(entityId));
        }
    }

    public static void clearSignal(AISignalData signal) {
        if (signal == null) {
            return;
        }
        List<AISignalData> typeSignals = activeSignals.get(signal.getType());
        if (typeSignals != null) {
            typeSignals.remove(signal);
        }
        List<AISignalData> entitySignalsList = entitySignals.get(signal.getSourceEntityId());
        if (entitySignalsList != null) {
            entitySignalsList.remove(signal);
        }
    }
}

import java.util.function.Predicate;

/**
 * Exhaustive spawn-location search loop (O(n³); suitability probes supplied by L3).
 */
public final class SpawnLocationSearchPolicy {

    private SpawnLocationSearchPolicy() {}

    public static BlockPosRef findFirstSuitable(
            BlockPosRef origin, int searchRadius, int maxHeight, Predicate<BlockPosRef> suitabilityProbe) {
        if (origin == null || suitabilityProbe == null || searchRadius <= 0 || maxHeight <= 0) {
            return null;
        }

        for (int x = -searchRadius; x <= searchRadius; x++) {
            for (int z = -searchRadius; z <= searchRadius; z++) {
                for (int y = -maxHeight; y <= maxHeight; y++) {
                    BlockPosRef candidate = origin.offset(x, y, z);
                    if (suitabilityProbe.test(candidate)) {
                        return candidate;
                    }
                }
            }
        }
        return null;
    }
}

/**
 * Spawn policy for specimen pod release actions.
 */
public final class SpecimenPodSpawnPolicy {

    private SpecimenPodSpawnPolicy() {
    }

    public record SpawnRequest(String specimenType, boolean structureFormed, boolean hasSpecimen, SpawnContext context) {
    }

    public static boolean canSpawn(SpawnRequest request) {
        if (request == null || !request.structureFormed() || !request.hasSpecimen()) {
            return false;
        }
        return resolveForm(request.specimenType()).isPresent();
    }

    public static Optional<XenomorphType> resolveForm(String specimenType) {
        if (specimenType == null || specimenType.isBlank() || "none".equalsIgnoreCase(specimenType)) {
            return Optional.empty();
        }
        return switch (specimenType.toLowerCase(Locale.ROOT)) {
            case "facehugger" -> Optional.of(XenomorphType.FACEHUGGER);
            case "chestburster" -> Optional.of(XenomorphType.CHESTBURSTER);
            default -> Optional.empty();
        };
    }

    public static boolean spawnSpecimen(EntitySpawnPort spawnPort, SpawnRequest request) {
        if (!canSpawn(request) || spawnPort == null) {
            return false;
        }
        XenomorphType form = resolveForm(request.specimenType()).orElse(null);
        if (form == null) {
            return false;
        }
        return spawnPort.spawnForm(form, request.context());
    }
}

            breakStructure(port);
        }
    }

    public static boolean validateStructure(SpecimenPodPort port) {
        port.logDebug("Validating specimen pod structure at " + port.controllerPosition());

        if (!port.isControllerBlock()) {
            port.setStructureState(port.structureState().transitionOnValidate(false));
            port.markDirty();
            return false;
        }

        SpecimenPodBlueprint.PodValidationResult result = SpecimenPodBlueprint.validate(
                port::isOccupied);
        port.applyValidationResult(result);
        return result.isValid();
    }

    public static void breakStructure(SpecimenPodPort port) {
        port.logInfo("Breaking specimen pod structure at " + port.controllerPosition());

        for (BlockPosRef pos : port.structureBlocks()) {
            if (port.isControllerAt(pos)) {
                continue;
            }
            port.destroyBlock(pos);
        }

        port.setStructureBlocks(List.of());
        port.setStructureState(port.structureState().transitionOnBreak());
        port.clearSpecimenData();
        port.markDirty();
    }

    public static List<BlockPosRef> offsetsToWorld(BlockPosRef center, List<SpecimenPodBlueprint.BlockOffset> offsets) {
        List<BlockPosRef> world = new ArrayList<>(offsets.size());
        for (SpecimenPodBlueprint.BlockOffset offset : offsets) {
            world.add(center.offset(offset.x(), offset.y(), offset.z()));
        }
        return world;
    }

    private static int validationIntervalTicks() {
        LoadedBlockVisualConfig config = BlockConfigLoader.getConfig(CONTROLLER_BLOCK);
        if (config == null || config.validationIntervalTicks <= 0) {
            throw new ConfigContractException("SpecimenPodWorkflow",
                    "/data/weylandyutani/blocks/" + CONTROLLER_BLOCK + ".json",
                    "gameplay.validation_interval_ticks", "required positive integer");
        }
        return config.validationIntervalTicks;
    }

import com.weylandyutani.layer_1_infrastructure.level_3.territory.TerritoryConfig;

import java.util.ArrayList;
import java.util.List;

/**
 * Space validation logic for hive expansion.
 */
public class ExpansionValidator {

    private static final int GROUND_LEVEL_OFFSET = 0;

    private ExpansionValidator() {
    }

    private static double calculateDistance2D(int x, int z) {
        return Math.sqrt(x * x + z * z);
    }

    public static ValidationResult canExpandToHive(
            WorldQueryPort query, WorldRef world, BlockPosRef centerPos, int targetRadius) {
        if (query == null || world == null || centerPos == null) {
            return ValidationResult.failure("Cannot validate without server world context");
        }
        int nestInnerRadius = TerritoryConfig.getNestInnerRadius();
        List<BlockPosRef> obstacles = scanExpansionArea(query, world, centerPos, nestInnerRadius, targetRadius);
        double obstaclePercentage = getObstaclePercentage(query, world, centerPos, targetRadius);

        double requiredClearance = TerritoryConfig.getExpansionClearanceRequired() / 100.0;
        if (obstaclePercentage > (1.0 - requiredClearance)) {
            return ValidationResult.failure(
                    String.format("Insufficient clearance: %.1f%% obstacles (need %.1f%% clear)",
                            obstaclePercentage * 100, requiredClearance * 100));
        }

        for (BlockPosRef obstacle : obstacles) {
            String blockId = query.blockIdAt(world, obstacle);
            if (isUnbreakableBlock(blockId)) {
                return ValidationResult.failure("Unbreakable obstacles detected: " + blockId);
            }
        }

        ModLogger.info("ExpansionValidator",
                String.format("✅ Expansion validation passed: %.1f%% clear space",
                        (1.0 - obstaclePercentage) * 100));

        return ValidationResult.success();
    }

    public static List<BlockPosRef> scanExpansionArea(
            WorldQueryPort query,
            WorldRef world,
            BlockPosRef centerPos,
            int innerRadius,
            int outerRadius) {
        List<BlockPosRef> obstacles = new ArrayList<>();

        for (int x = -outerRadius; x <= outerRadius; x++) {
            for (int z = -outerRadius; z <= outerRadius; z++) {
                if (Math.abs(x) <= innerRadius && Math.abs(z) <= innerRadius) {
                    continue;
                }

                double distance = calculateDistance2D(x, z);
                if (distance > innerRadius && distance <= outerRadius) {
                    BlockPosRef pos = centerPos.offset(x, GROUND_LEVEL_OFFSET, z);

                    if (hasObstacleAt(query, world, pos) || hasObstacleAt(query, world, pos.above())) {
                        obstacles.add(pos);
                    }
                }
            }
        }

        return obstacles;
    }

    public static double getObstaclePercentage(
            WorldQueryPort query, WorldRef world, BlockPosRef centerPos, int radius) {
        int totalPositions = 0;
        int obstaclePositions = 0;
        int nestInnerRadius = TerritoryConfig.getNestInnerRadius();

        for (int x = -radius; x <= radius; x++) {
            for (int z = -radius; z <= radius; z++) {
                if (Math.abs(x) <= nestInnerRadius && Math.abs(z) <= nestInnerRadius) {
                    continue;
                }

                double distance = calculateDistance2D(x, z);
                if (distance > nestInnerRadius && distance <= radius) {
                    totalPositions++;

                    BlockPosRef pos = centerPos.offset(x, GROUND_LEVEL_OFFSET, z);
                    if (hasObstacleAt(query, world, pos) || hasObstacleAt(query, world, pos.above())) {
                        obstaclePositions++;
                    }
                }
            }
        }

        return totalPositions > 0 ? (double) obstaclePositions / totalPositions : 0.0;
    }

    private static boolean hasObstacleAt(WorldQueryPort query, WorldRef world, BlockPosRef pos) {
        if (!query.isInBounds(world, pos)) {
            return true;
        }

        if (query.isAirAt(world, pos)) {
            return false;
        }

        String blockId = query.blockIdAt(world, pos);
        if (isUnbreakableBlock(blockId)) {
            return true;
        }

        return query.canOccludeAt(world, pos);
    }

    private static boolean isUnbreakableBlock(String blockId) {
        return UnbreakableBlockIds.contains(blockId); //TODO: we no longer use this and it is data driven. this needs updated.
    }
}

        if (!nearby.isEmpty()) {

            if (!territory.getQueenUUID().equals(selfQueenId)) {
                return true;
            }
        }
        return false;
    }

        if (query == null || world == null || pos == null) {
            return false;
        }
        if (!query.canOccludeAt(world, pos.below())) {
            return false;
        }
        if (!query.isAirAt(world, pos)) {
            return false;
        }
        return query.isAirAt(world, pos.above());

        }

        if (!hasSuitableGround(query, world, pos)) {
            return ValidationResult.failure("Ground is not suitable for nest building");
        }

        return ValidationResult.success();
    }

/**
 * Targeting policy for xenomorph combat goals: valid hosts and priority ranking.
 */
public final class XenomorphTargetingHelper {

    private XenomorphTargetingHelper() {
        throw new UnsupportedOperationException("XenomorphTargetingHelper is a utility class");
    }

    public static Predicate<LivingEntityRef> isValidTarget() {
        return XenomorphTargetingHelper::isValidTargetEntity;
    }

    public static boolean isValidTargetEntity(LivingEntityRef entity) {
        if (!BoundaryServices.isInitialized()) {
            return false;
        }
        EntityQueryPort query = BoundaryServices.entityQuery();
        if (entity == null || !query.isValid(entity)) {
            return false;
        }
        if (query.classify(entity) == EntityKind.MOD_XENOMORPH) {
            return false;
        }
        if (query.classify(entity) == EntityKind.PLAYER && query.isCreativePlayer(entity)) {
            return false;
        }
        String mobName = query.encodeId(entity).map(XenomorphTargetingHelper::registryPath).orElse(null);
        return mobName != null && MobHostRegistry.isSuitableForXenomorph(mobName);
    }

    public static int getTargetPriority(LivingEntityRef entity) {
        if (!BoundaryServices.isInitialized() || entity == null) {
            return 7;
        }
        return switch (BoundaryServices.entityQuery().classify(entity)) {
            case PLAYER -> 2;
            case VILLAGER -> 3;
            case ANIMAL -> 4;
            default -> 7;
        };
    }

    public record ScoredTarget(LivingEntityRef ref, double distanceSq) {
    }

    public static void addScoredCandidate(
            java.util.List<ScoredTarget> candidates,
            LivingEntityRef ref,
            double distanceSq) {
        if (isValidTargetEntity(ref)) {
            candidates.add(new ScoredTarget(ref, distanceSq));
        }
    }

    /**
     * Selects the highest-priority target; ties break on nearest distance.
     */
    public static LivingEntityRef selectBestTarget(java.util.List<ScoredTarget> candidates) {
        LivingEntityRef best = null;
        int bestPriority = Integer.MAX_VALUE;
        double bestDistanceSq = Double.MAX_VALUE;

        for (ScoredTarget candidate : candidates) {
            if (candidate == null || candidate.ref() == null) {
                continue;
            }
            int priority = getTargetPriority(candidate.ref());
            double distanceSq = candidate.distanceSq();
            if (priority < bestPriority || (priority == bestPriority && distanceSq < bestDistanceSq)) {
                best = candidate.ref();
                bestPriority = priority;
                bestDistanceSq = distanceSq;
            }
        }
        return best;
    }

    private static String registryPath(String encodeId) {
        if (encodeId == null || encodeId.isBlank()) {
            return null;
        }
        int separator = encodeId.indexOf(':');
        return separator >= 0 ? encodeId.substring(separator + 1) : encodeId;
    }
}

            return false;

                continue;

                    BoundaryServices.worldMutation(), world, checkPos, blockId, actorLabel)) {
                ModLogger.debug("BlockCombatHelper", "🔨 Broke occluding block: " + blockId + " at " + checkPos);

    }

import com.weylandyutani.layer_0_domain.level_0.effects.DomainEffectKey;

import com.weylandyutani.layer_2_systems.level_0.services.BoundaryServices;

/**
 * Focused manager for gestation effect logic.
 *
 * FILE STATUS: STABLE
 */
public class GestationManager {

    public static void applyGestationEffect(LivingEntityRef host) {
        if (EntityValidationHelper.isInvalid(host, BoundaryServices.entityQuery()) || !BoundaryServices.isInitialized()) {
            ModLogger.info("Skipping gestation application - host is invalid or boundary not ready");
            return;
        }

        if (BoundaryServices.effects().hasEffect(host, DomainEffectKey.GESTATION)) {
            ModLogger.info("Skipping gestation application - host already has effect");
            return;
        }

        int gestationDuration = getGestationDuration();
        if (BoundaryServices.entityQuery().isPlayer(host)) {
            gestationDuration *= 3;
        }

        BoundaryServices.effects().apply(host, DomainEffectKey.GESTATION, gestationDuration, 0,
                new EffectApplyOptions(false, true, true));

        if (BoundaryServices.effects().hasEffect(host, DomainEffectKey.GESTATION)) {
            ModLogger.info("Gestation effect applied to "
                    + BoundaryServices.presentation().displayName(host)
                    + " for " + gestationDuration + " ticks");
        } else {
            ModLogger.error("Failed to apply gestation effect to "
                    + BoundaryServices.presentation().displayName(host));
        }
    }

    public static boolean handleGestationTick(LivingEntityRef host, int timeRemaining) {
        if (!BoundaryServices.isInitialized()) {
            return false;
        }

        if (timeRemaining < 0) {
            ModLogger.error("Invalid gestation duration: " + timeRemaining + " ticks for "
                    + BoundaryServices.presentation().displayName(host));
            return false;
        }

        if (timeRemaining % 200 == 0) {
            ModLogger.info("Gestation progress for "
                    + BoundaryServices.presentation().displayName(host)
                    + ": " + timeRemaining + " ticks remaining");
        }

        int warningPhaseStart = getWarningPhaseStart();
        int finalPhaseStart = getFinalPhaseStart();

        if (timeRemaining == warningPhaseStart) {
            ModLogger.info("Gestation warning phase starting for "
                    + BoundaryServices.presentation().displayName(host)
                    + " - " + timeRemaining + " ticks remaining");
        }

        if (timeRemaining == finalPhaseStart) {
            ModLogger.info("Gestation final phase starting for "
                    + BoundaryServices.presentation().displayName(host)
                    + " - " + timeRemaining + " ticks remaining");
        }

        if (timeRemaining <= warningPhaseStart && timeRemaining > finalPhaseStart) {
            handleWarningPhase(host, timeRemaining);
        }

        if (timeRemaining <= finalPhaseStart) {
            handleFinalPhase(host);
            return false;
        }

        return true;
    }

    private static void handleWarningPhase(LivingEntityRef host, int timeRemaining) {
        int damageInterval = getWarningDamageInterval();
        float damageAmount = getWarningDamageAmount();
        int particleCount = getWarningParticleCount();
        float particleSpread = getWarningParticleSpread();

        if (timeRemaining % damageInterval == 0) {
            BoundaryServices.presentation().applyMagicDamage(host, damageAmount);
        }

        BoundaryServices.presentation().spawnDamageIndicatorParticles(host, particleCount, particleSpread);
    }

    private static void handleFinalPhase(LivingEntityRef host) {
        ModLogger.info("Gestation complete! Chestburster emerging from "
                + BoundaryServices.presentation().displayName(host));

        spawnChestburster(host);

        float finalDamage = getFinalDamageAmount();
        ModLogger.info("Killing host " + BoundaryServices.presentation().displayName(host)
                + " with " + finalDamage + " damage");
        BoundaryServices.presentation().applyMagicDamage(host, finalDamage);

        ModLogger.info("Gestation process completed for "
                + BoundaryServices.presentation().displayName(host));
    }

    private static void spawnChestburster(LivingEntityRef host) {
        if (!BoundaryServices.isInitialized()) {
            ModLogger.warn("BoundaryServices not initialized — cannot spawn chestburster");
            return;
        }

        try {
            BoundaryServices.entitySpawn().spawnXenomorph(
                    BoundaryServices.presentation().spawnContextAt(host),
                    XenomorphType.CHESTBURSTER);

            BoundaryServices.presentation().spawnExplosionParticles(
                    host, getExplosionParticleCount(), getExplosionParticleSpread());

            ModLogger.info("Chestburster spawned successfully for host "
                    + BoundaryServices.presentation().displayName(host));
        } catch (Exception e) {
            ModLogger.error("Failed to spawn chestburster: " + e.getMessage(), e);
        }
    }

    public static boolean isGestating(LivingEntityRef entity) {
        return entity != null && BoundaryServices.isInitialized()
                && BoundaryServices.effects().hasEffect(entity, DomainEffectKey.GESTATION);
    }

    public static int getRemainingGestationTime(LivingEntityRef entity) {
        if (!isGestating(entity)) {
            return 0;
        }
        return BoundaryServices.effects().getRemainingDuration(entity, DomainEffectKey.GESTATION);
    }

    private static int getGestationDuration() { return 1200; }
    private static int getWarningPhaseStart() { return 1200; }
    private static int getFinalPhaseStart() { return 1200; }
    private static int getWarningDamageInterval() { return 1200; }
    private static float getWarningDamageAmount() { return 1.0f; }
    private static int getWarningParticleCount() { return 1200; }
    private static float getWarningParticleSpread() { return 1.0f; }
    private static float getFinalDamageAmount() { return 1.0f; }
    private static int getExplosionParticleCount() { return 1200; }
    private static float getExplosionParticleSpread() { return 1.0f; }
}

package com.weylandyutani.layer_2_systems.level_1.hive;

import com.weylandyutani.layer_0_domain.level_0.spatial.BlockPosRef;
import com.weylandyutani.layer_2_systems.level_0.hive.NestBuildingPort;
import com.weylandyutani.layer_2_systems.level_0.hive.NestBuildingTiming;

/**
 * Orchestration workflow for nest building lifecycle.
 * Encapsulates timing, phase progression, and hive expansion logic.
 * 
 * Boundaries:
 * - Does NOT import Minecraft types
 * - Does NOT manage territory (delegates to port)
 * - Does NOT perform block placement (delegates to port)
 * - Manages timing and phase progression state
 */
public final class NestBuildingWorkflow {
    private final NestBuildingPort port;
    private final NestBuildingTiming timing;
    private int buildingProgress;
    private int hiveBuildingProgress;
    private int phaseDelayTicks;
    private int cooldownTicks;
    private boolean expandingToHive;

    public NestBuildingWorkflow(NestBuildingPort port, NestBuildingTiming timing) {
        this.port = port;
        this.timing = timing;
    }

    /**
     * Advance the nest building workflow by one tick.
     * Handles cooldown, phase delays, nest progression, and hive expansion.
     */
    public void tick() {
        // Cooldown period after nest creation
        if (this.cooldownTicks > 0) {
            --this.cooldownTicks;
            return;
        }
        
        // Delay between phases
        if (this.phaseDelayTicks > 0) {
            --this.phaseDelayTicks;
            return;
        }
        
        // Build nest structure phases
        if (this.buildingProgress < this.timing.nestBuildingPhases()) {
            if (this.port.buildNestStructurePhase(this.buildingProgress)) {
                ++this.buildingProgress;
                this.phaseDelayTicks = this.timing.phaseBuildDelayTicks();
            }
            return;
        }
        
        // Check if ready to expand to hive
        if (!this.expandingToHive && this.port.checkTerritoryProgression()) {
            this.expandingToHive = true;
            this.hiveBuildingProgress = 0;
        }
        
        // Build hive structure phases
        if (this.expandingToHive && this.hiveBuildingProgress < this.timing.hiveBuildingPhases()) {
            if (this.port.buildHiveStructurePhase(this.hiveBuildingProgress)) {
                ++this.hiveBuildingProgress;
                this.phaseDelayTicks = this.timing.phaseBuildDelayTicks();
            }
            if (this.hiveBuildingProgress >= this.timing.hiveBuildingPhases()) {
                this.port.onHiveStructureComplete();
            }
        }
    }

    /**
     * Restore workflow state from saved data.
     */
    public void restoreFromSave(BlockPosRef nestCenter, int savedBuildingProgress, int savedHiveProgress, boolean savedExpanding) {
        this.port.restoreNestState(nestCenter, savedBuildingProgress, savedHiveProgress, savedExpanding);
        this.buildingProgress = savedBuildingProgress;
        this.hiveBuildingProgress = savedHiveProgress;
        this.expandingToHive = savedExpanding;
        this.cooldownTicks = this.timing.nestCreationCooldownTicks();
    }

    public int getBuildingProgress() {
        return this.buildingProgress;
    }

    public boolean isNestComplete() {
        return this.buildingProgress >= this.timing.nestBuildingPhases();
    }

    public int getHiveBuildingProgress() {
        return this.hiveBuildingProgress;
    }

    public boolean isExpandingToHive() {
        return this.expandingToHive;
    }

    public boolean isHiveExpansionComplete() {
        return this.expandingToHive && this.hiveBuildingProgress >= this.timing.hiveBuildingPhases();
    }
}

/**
 * Central coordinator for entity maturation lifecycle keyed by {@link XenomorphType}.
 *
 * FILE STATUS: STABLE
 */
public class MaturationSystem {
    private static final Map<XenomorphType, MaturationHandler> PROCESS_REGISTRY = new HashMap<>();
    private static final Map<XenomorphType, ToIntFunction<LivingEntityRef>> TIMER_GETTER_REGISTRY = new HashMap<>();
    private static final Map<XenomorphType, ObjIntConsumer<LivingEntityRef>> TIMER_SETTER_REGISTRY = new HashMap<>();
    private static final Map<XenomorphType, ObjIntConsumer<LivingEntityRef>> TIMER_REDUCER_REGISTRY = new HashMap<>();
    private static final Map<XenomorphType, Function<LivingEntityRef, Boolean>> FOOD_HUNTING_REGISTRY = new HashMap<>();
    private static final Map<XenomorphType, ToDoubleFunction<LivingEntityRef>> URGENCY_REGISTRY = new HashMap<>();
    private static final Map<XenomorphType, ToIntFunction<LivingEntityRef>> TIMER_REDUCTION_REGISTRY = new HashMap<>();

    private static void registerMaturationHandlers() {
        registerType(XenomorphType.CHESTBURSTER,
                MaturationSystem::advanceMaturationTimer,
                MaturationSystem::reduceMaturationTimerGeneric,
                MaturationSystem::shouldPrioritizeFoodHuntingGeneric,
                MaturationSystem::getMaturationUrgencyGeneric,
                ref -> LifecycleConfig.getDefaultFoodTimerReductionTicks());

        registerType(XenomorphType.DRONE,
                MaturationSystem::advanceMaturationTimer,
                MaturationSystem::reduceMaturationTimerGeneric,
                MaturationSystem::shouldPrioritizeFoodHuntingGeneric,
                MaturationSystem::getMaturationUrgencyGeneric,
                ref -> LifecycleConfig.getDefaultFoodTimerReductionTicks());
    }

    private static void advanceMaturationTimer(@Nonnull LivingEntityRef entity) {
        EntityConfig config = maturationConfig(entity);
        if (config == null || config.maturationTime <= 0) {
            return;
        }
        int timer = getMaturationTimer(entity);
        if (timer < config.maturationTime) {
            setMaturationTimer(entity, timer + 1);
        }
    }

    private static void reduceMaturationTimerGeneric(@Nonnull LivingEntityRef entity, int reduction) {
        if (reduction <= 0) {
            return;
        }
        int timer = getMaturationTimer(entity);
        setMaturationTimer(entity, Math.max(0, timer - reduction));
    }

    private static boolean shouldPrioritizeFoodHuntingGeneric(@Nonnull LivingEntityRef entity) {
        return getMaturationUrgencyGeneric(entity) < LifecycleConfig.getFoodHuntUrgencyThreshold();
    }

    private static double getMaturationUrgencyGeneric(@Nonnull LivingEntityRef entity) {
        EntityConfig config = maturationConfig(entity);
        if (config == null || config.maturationTime <= 0) {
            return 1.0;
        }
        int timer = getMaturationTimer(entity);
        return Math.min(1.0, (double) timer / config.maturationTime);
    }

    @Nullable
    private static EntityConfig maturationConfig(@Nonnull LivingEntityRef entity) {
        if (shouldSkipEntity(entity)) {
            return null;
        }
        XenomorphType type = resolveType(entity);
        if (type == null) {
            return null;
        }
        EntityConfig config = EntityConfigStorage.getConfig(type);
        if (config == null || !config.canMature) {
            return null;
        }
        return config;
    }

    private static void registerType(XenomorphType type,
                                     MaturationHandler processor,
                                     ObjIntConsumer<LivingEntityRef> timerReducer,
                                     Function<LivingEntityRef, Boolean> foodHunting,
                                     ToDoubleFunction<LivingEntityRef> urgency,
                                     ToIntFunction<LivingEntityRef> timerReduction) {
        PROCESS_REGISTRY.put(type, processor);
        TIMER_GETTER_REGISTRY.put(type, MaturationSystem::readMaturationTimer);
        TIMER_SETTER_REGISTRY.put(type, MaturationSystem::writeMaturationTimer);
        TIMER_REDUCER_REGISTRY.put(type, timerReducer);
        FOOD_HUNTING_REGISTRY.put(type, foodHunting);
        URGENCY_REGISTRY.put(type, urgency);
        TIMER_REDUCTION_REGISTRY.put(type, timerReduction);
    }

    public static void registerEntityType(@Nonnull XenomorphType type,
                                          @Nonnull MaturationHandler processor,
                                          @Nonnull ToIntFunction<LivingEntityRef> timerGetter,
                                          @Nonnull ObjIntConsumer<LivingEntityRef> timerSetter,
                                          @Nonnull ObjIntConsumer<LivingEntityRef> timerReducer,
                                          @Nonnull Function<LivingEntityRef, Boolean> foodHunting,
                                          @Nonnull ToDoubleFunction<LivingEntityRef> urgency,
                                          @Nonnull ToIntFunction<LivingEntityRef> timerReduction) {
        PROCESS_REGISTRY.put(type, processor);
        TIMER_GETTER_REGISTRY.put(type, timerGetter);
        TIMER_SETTER_REGISTRY.put(type, timerSetter);
        TIMER_REDUCER_REGISTRY.put(type, timerReducer);
        FOOD_HUNTING_REGISTRY.put(type, foodHunting);
        URGENCY_REGISTRY.put(type, urgency);
        TIMER_REDUCTION_REGISTRY.put(type, timerReduction);
    }

    public static void processMaturation(@Nonnull LivingEntityRef entity) {
        if (shouldSkipEntity(entity)) {
            return;
        }
        MaturationHandler handler = PROCESS_REGISTRY.get(resolveType(entity));
        if (handler != null) {
            handler.process(entity);
        }
    }

    public static int resolveMaturationTimer(@Nonnull LivingEntityRef entity) {
        return readMaturationTimer(entity);
    }

    public static void resolveSetMaturationTimer(@Nonnull LivingEntityRef entity, int timer) {
        writeMaturationTimer(entity, timer);
    }

    private static int readMaturationTimer(@Nonnull LivingEntityRef entity) {
        if (MaturationServices.isInitialized()) {
            return MaturationServices.maturation().getMaturationTimer(entity);
        }
        return 0;
    }

    private static void writeMaturationTimer(@Nonnull LivingEntityRef entity, int timer) {
        if (MaturationServices.isInitialized()) {
            MaturationServices.maturation().setMaturationTimer(entity, timer);
        }
    }

    public static int getMaturationTimer(@Nonnull LivingEntityRef entity) {
        if (shouldSkipEntity(entity)) {
            return 0;
        }
        ToIntFunction<LivingEntityRef> getter = TIMER_GETTER_REGISTRY.get(resolveType(entity));
        return getter != null ? getter.applyAsInt(entity) : 0;
    }

    public static void setMaturationTimer(@Nonnull LivingEntityRef entity, int timer) {
        if (shouldSkipEntity(entity)) {
            return;
        }
        ObjIntConsumer<LivingEntityRef> setter = TIMER_SETTER_REGISTRY.get(resolveType(entity));
        if (setter != null) {
            setter.accept(entity, timer);
        }
    }

    public static void reduceMaturationTimer(@Nonnull LivingEntityRef entity, int reduction) {
        if (shouldSkipEntity(entity)) {
            return;
        }
        ObjIntConsumer<LivingEntityRef> reducer = TIMER_REDUCER_REGISTRY.get(resolveType(entity));
        if (reducer != null) {
            reducer.accept(entity, reduction);
        }
    }

    public static boolean shouldPrioritizeFoodHunting(@Nonnull LivingEntityRef entity) {
        if (shouldSkipEntity(entity)) {
            return false;
        }
        Function<LivingEntityRef, Boolean> handler = FOOD_HUNTING_REGISTRY.get(resolveType(entity));
        return handler != null && handler.apply(entity);
    }

    public static double getMaturationUrgency(@Nonnull LivingEntityRef entity) {
        if (shouldSkipEntity(entity)) {
            return 0.0;
        }
        ToDoubleFunction<LivingEntityRef> handler = URGENCY_REGISTRY.get(resolveType(entity));
        return handler != null ? handler.applyAsDouble(entity) : 0.0;
    }

    public static int getFoodTimerReduction(@Nonnull LivingEntityRef entity) {
        if (shouldSkipEntity(entity)) {
            return 0;
        }
        ToIntFunction<LivingEntityRef> handler = TIMER_REDUCTION_REGISTRY.get(resolveType(entity));
        return handler != null ? handler.applyAsInt(entity) : 0;
    }

    private static boolean shouldSkipEntity(@Nonnull LivingEntityRef entity) {
        return !BoundaryServices.isInitialized()
                || EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery());
    }

    @Nullable
    private static XenomorphType resolveType(@Nonnull LivingEntityRef entity) {
        if (!BoundaryServices.isInitialized()) {
            return null;
        }
        return BoundaryServices.formQuery().resolveForm(entity);
    }

    static {
        registerMaturationHandlers();
    }
}

/**
 * Data-driven manager for host effects during facehugger attachment.
 *
 * FILE STATUS: FUNCTIONAL
 */
public class HostEffectManager {

    private static final Set<UUID> warnedEntities = new HashSet<>();

    public static void applyHostImmobilization(LivingEntityRef host) {
        if (EntityValidationHelper.isInvalid(host, BoundaryServices.entityQuery()) || !BoundaryServices.isInitialized()) {
            return;
        }

        String entityName = BoundaryServices.entityQuery().encodeId(host)
                .map(HostEffectManager::registryPath)
                .orElse("");
        HostType hostType = MobHostRegistry.getHostType(entityName);
        if (hostType == null) {
            ModLogger.warn("Unknown entity type '" + entityName + "' - defaulting to LIVING_BIOLOGICAL");
            hostType = HostType.LIVING_BIOLOGICAL;
        }

        HostEffectConfig.HostTypeConfig config = HostEffectConfig.getHostTypeConfig(hostType);
        if (!config.canBeImmobilized) {
            logUnusualHost(host, hostType, "Cannot be immobilized");
            return;
        }

        applyVisualEffects(host, config);
        applyMovementEffects(host, config);

        if (config.requiresPlayerSpecificEffects && BoundaryServices.entityQuery().isPlayer(host)) {
            BoundaryServices.presentation().applyPlayerAttachmentPose(host);
        }

        if (BoundaryServices.presentation().tickCount(host) % 200 == 0) {
            ModLogger.info("Host " + BoundaryServices.presentation().displayName(host)
                    + " (" + hostType.getKey() + ") is being facehugged - immobilized");
        }
    }

    private static void applyVisualEffects(LivingEntityRef host, HostEffectConfig.HostTypeConfig config) {
        applyEffectList(host, config.visualEffects);
    }

    private static void applyMovementEffects(LivingEntityRef host, HostEffectConfig.HostTypeConfig config) {
        applyEffectList(host, config.movementEffects);
        applyUniversalMovementRestrictions(host);
    }

    private static void applyEffectList(LivingEntityRef host, List<HostEffectConfig.EffectConfig> effects) {
        var effectPort = BoundaryServices.effects();
        for (HostEffectConfig.EffectConfig effectConfig : effects) {
            if (effectConfig.standardEffect != null) {
                if (effectPort.hasStandardEffect(host, effectConfig.standardEffect)) {
                    continue;
                }
                effectPort.applyStandard(host, effectConfig.standardEffect, effectConfig.duration,
                        effectConfig.amplifier, effectConfig.toOptions());
            } else if (effectConfig.domainEffect != null) {
                if (effectPort.hasEffect(host, effectConfig.domainEffect)) {
                    continue;
                }
                effectPort.apply(host, effectConfig.domainEffect, effectConfig.duration,
                        effectConfig.amplifier, effectConfig.toOptions());
            }
        }
    }

    private static void applyUniversalMovementRestrictions(LivingEntityRef host) {
        BoundaryServices.presentation().setDeltaMovement(host, 0, 0, 0);
        BoundaryServices.presentation().releaseUsingItem(host);
    }

    public static void removeHostEffects(LivingEntityRef host) {
        if (EntityValidationHelper.isInvalid(host, BoundaryServices.entityQuery()) || !BoundaryServices.isInitialized()) {
            return;
        }

        removeAllPossibleEffects(host);

        if (BoundaryServices.entityQuery().isPlayer(host)) {
            BoundaryServices.presentation().restorePlayerAttachmentPose(host);
        }

        ModLogger.info("Host effects removed from " + BoundaryServices.presentation().displayName(host));
    }

    private static void removeAllPossibleEffects(LivingEntityRef host) {
        var effectPort = BoundaryServices.effects();
        for (StandardMobEffectKey key : StandardMobEffectKey.values()) {
            effectPort.removeStandard(host, key);
        }
        effectPort.remove(host, DomainEffectKey.FACEHUGGER_ATTACHMENT);
    }

    public static boolean isHostBeingFacehugged(LivingEntityRef entity) {
        if (entity == null || !BoundaryServices.isInitialized()) {
            return false;
        }
        return BoundaryServices.effects().hasStandardEffect(entity, StandardMobEffectKey.BLINDNESS);
    }

    /**
     * Mob-effect refresh duration (ticks) while a host is facehugged.
     * Authority: {@code facehugger.json} {@code attachment_parameters.effect_refresh_ticks}
     * via {@link AttachmentConfigHelper} (aligned with {@code lifecycle.json} {@code attachment.effect_refresh_ticks}).
     */
    public static int getHostEffectDuration() {
        return AttachmentConfigHelper.getAttachmentEffectRefreshTicks();
    }

    private static void logUnusualHost(LivingEntityRef host, HostType hostType, String reason) {
        UUID entityId = host.entityId();
        if (!warnedEntities.contains(entityId)) {
            ModLogger.warn("FACEHUGGER_ATTACHMENT effect applied to unusual host: "
                    + BoundaryServices.presentation().displayName(host)
                    + " (" + hostType.getKey() + ") - " + reason);
            warnedEntities.add(entityId);
        }
    }

    private static String registryPath(String encodeId) {
        if (encodeId == null || encodeId.isBlank()) {
            return "";
        }
        int separator = encodeId.indexOf(':');
        return separator >= 0 ? encodeId.substring(separator + 1) : encodeId;
    }
}

import java.util.UUID;

/**
 * Territory stage progression for queen nest-to-hive lifecycle.
 */
public final class TerritoryProgressionSystem {

    private TerritoryProgressionSystem() {
    }

    public static boolean checkProgression(WorldRef world, UUID queenUuid) {
        if (world == null || queenUuid == null) {
            return false;
        }
        QueenTerritory territory = TerritoryManager.getTerritory(queenUuid);
        if (territory == null) {
            return false;
        }
        return territory.getStage() == QueenTerritory.TerritoryStage.NEST
                && territory.getStructureStage() == QueenTerritory.StructureStage.NEST_STRUCTURE;
    }

    public static void updateStructureToHive(UUID queenUuid) {
        if (queenUuid == null) {
            return;
        }
        QueenTerritory territory = TerritoryManager.getTerritory(queenUuid);
        if (territory == null) {
            ModLogger.warn("TerritoryProgressionSystem", "No territory for queen: " + queenUuid);
            return;
        }
        QueenTerritory updated = new QueenTerritory(
                territory.getCenterPos(),
                TerritoryConfig.getHiveRadius(),
                QueenTerritory.TerritoryStage.HIVE,
                QueenTerritory.StructureStage.HIVE_STRUCTURE,
                queenUuid,
                territory.getCreationTimestamp()
        );
        TerritoryManager.registerTerritory(queenUuid, updated);
        ModLogger.info("TerritoryProgressionSystem", "Hive progression applied for queen: " + queenUuid);
    }
}

package com.weylandyutani.layer_2_systems.level_2.parasite;

import javax.annotation.Nonnull;
import javax.annotation.Nullable;

import com.weylandyutani.layer_0_domain.level_0.effects.DomainEffectKey;
import com.weylandyutani.layer_0_domain.level_0.effects.EffectApplyOptions;
import com.weylandyutani.layer_0_domain.level_0.parasite.AttachmentState;
import com.weylandyutani.layer_0_domain.level_1.entity.LivingEntityRef;
import com.weylandyutani.layer_0_domain.level_2.lifecycle.AttachmentStatePort;
import com.weylandyutani.layer_1_infrastructure.level_0.platform.ModLogger;
import com.weylandyutani.layer_1_infrastructure.level_1.entity.AttachmentConfigHelper;
import com.weylandyutani.layer_2_systems.level_0.entity.EntityValidationHelper;
import com.weylandyutani.layer_2_systems.level_0.services.AttachmentServices;
import com.weylandyutani.layer_2_systems.level_0.services.BoundaryServices;
import com.weylandyutani.layer_2_systems.level_1.parasite.FacehuggerSpentTransitionServices;
import com.weylandyutani.layer_2_systems.level_1.gestation.GestationManager;

/**
 * Focused manager for facehugger attachment logic.
 */
public class FacehuggerAttachmentManager {

    private static final String LOG = "FacehuggerAttachment";

    private FacehuggerAttachmentManager() {
    }

    public static boolean attemptAttachment(@Nonnull LivingEntityRef facehugger, @Nonnull LivingEntityRef host) {
        if (!supportsAttachment(facehugger) || EntityValidationHelper.isInvalid(host, BoundaryServices.entityQuery()) || !BoundaryServices.isInitialized()) {
            return false;
        }

        if (isSpent(facehugger)) {
            return false;
        }

        if (BoundaryServices.entityQuery().countAttachedFacehuggers(host) > 0) {
            ModLogger.info(LOG, "Host " + BoundaryServices.presentation().displayName(host)
                    + " already has an attached facehugger - denying attachment");
            return false;
        }

        if (isEntityBeingFacehugged(host)) {
            ModLogger.info(LOG, "Host " + BoundaryServices.presentation().displayName(host)
                    + " is already being facehugged");
            return false;
        }

        if (BoundaryServices.presentation().distance(facehugger, host) > AttachmentConfigHelper.getAttachmentDistance()) {
            return false;
        }

        startAttachment(facehugger, host);
        return true;
    }

    private static void startAttachment(@Nonnull LivingEntityRef facehugger, @Nonnull LivingEntityRef host) {
        ModLogger.info(LOG, "Facehugger " + BoundaryServices.presentation().displayName(facehugger)
                + " starting attachment to " + BoundaryServices.presentation().displayName(host)
                + " at distance " + String.format("%.2f", BoundaryServices.presentation().distance(facehugger, host)));

        setTarget(facehugger, host);
        setTicks(facehugger, 0);
        setState(facehugger, AttachmentState.ATTACHING);
        BoundaryServices.presentation().positionOnHost(facehugger, host, AttachmentConfigHelper.getFaceHeightRatio());

        BoundaryServices.presentation().setInvulnerable(facehugger, true);
        BoundaryServices.presentation().setRotation(facehugger, getHostYaw(host), 0.0f);
        applyAttachmentEffects(facehugger, host);

        ModLogger.info(LOG, "Facehugger attached to " + BoundaryServices.presentation().displayName(host)
                + " - immobile and invulnerable");
    }

    public static boolean handleAttachmentTick(@Nonnull LivingEntityRef facehugger, @Nonnull LivingEntityRef host,
                                               int attachmentTicks) {
        if (EntityValidationHelper.isInvalid(host, BoundaryServices.entityQuery())) {
            ModLogger.info(LOG, "Attachment tick " + attachmentTicks + ": Host is dead or null - ending attachment");
            BoundaryServices.presentation().applyMagicDamage(facehugger, Float.MAX_VALUE);
            return false;
        }

        if (BoundaryServices.presentation().distance(facehugger, host) > AttachmentConfigHelper.getAttachmentDistance()) {
            stopAttachment(facehugger, host);
            return false;
        }

        BoundaryServices.presentation().positionOnHost(facehugger, host, AttachmentConfigHelper.getFaceHeightRatio());

        int attachmentDuration = AttachmentConfigHelper.getAttachmentDuration();
        int implantationDelay = AttachmentConfigHelper.getImplantationDelay();
        int totalRequired = attachmentDuration + implantationDelay;

        if (attachmentTicks >= attachmentDuration) {
            if (attachmentTicks >= totalRequired) {
                ModLogger.info(LOG, "Implantation delay completed - proceeding with implantation");
                completeImplantation(facehugger, host);
                return false;
            }
            int remainingDelay = totalRequired - attachmentTicks;
            if (attachmentTicks % 40 == 0) {
                ModLogger.info(LOG, "Facehugger " + BoundaryServices.presentation().displayName(facehugger)
                        + " holding on to " + BoundaryServices.presentation().displayName(host)
                        + " - " + remainingDelay + " ticks until implantation");
            }
        }

        return true;
    }

    private static void completeImplantation(@Nonnull LivingEntityRef facehugger, @Nonnull LivingEntityRef host) {
        ModLogger.info(LOG, "Facehugger " + BoundaryServices.presentation().displayName(facehugger)
                + " completed implantation on " + BoundaryServices.presentation().displayName(host));
        GestationManager.applyGestationEffect(host);
        ModLogger.info(LOG, "Gestation effect applied to " + BoundaryServices.presentation().displayName(host));
        detachFacehugger(facehugger, host);
    }

    private static void detachFacehugger(@Nonnull LivingEntityRef facehugger, @Nonnull LivingEntityRef host) {
        ModLogger.info(LOG, "Starting facehugger detachment process");
        removeAttachmentEffect(host);
        removeImplantationEffect(facehugger);

        BoundaryServices.presentation().setNoGravity(facehugger, false);
        BoundaryServices.presentation().setInvulnerable(facehugger, false);
        BoundaryServices.presentation().setDeltaMovement(facehugger, 0, 0, 0);

        FacehuggerSpentTransitionServices.spentTransition().initializeSpentSession(facehugger);

        ModLogger.info(LOG, "Facehugger " + BoundaryServices.presentation().displayName(facehugger)
                + " marked as spent - will wander before dying");
    }

    private static void stopAttachment(@Nonnull LivingEntityRef facehugger, @Nullable LivingEntityRef host) {
        if (host != null) {
            removeAttachmentEffect(host);
        }
        removeImplantationEffect(facehugger);
        BoundaryServices.presentation().setNoGravity(facehugger, false);
        BoundaryServices.presentation().setInvulnerable(facehugger, false);
    }

    public static boolean isEntityBeingFacehugged(@Nonnull LivingEntityRef entity) {
        return entity != null && BoundaryServices.isInitialized()
                && BoundaryServices.effects().hasEffect(entity, DomainEffectKey.FACEHUGGER_ATTACHMENT);
    }

    public static int getAttachmentDuration() {
        return AttachmentConfigHelper.getAttachmentDuration();
    }

    public static double getAttachmentDistance() {
        return AttachmentConfigHelper.getAttachmentDistance();
    }

    public static boolean handleAttachmentStateBusinessLogic(@Nonnull LivingEntityRef facehugger,
                                                               @Nonnull LivingEntityRef host) {
        if (!supportsAttachment(facehugger) || EntityValidationHelper.isInvalid(host, BoundaryServices.entityQuery())) {
            return false;
        }

        return switch (getState(facehugger)) {
            case READY_TO_ATTACH -> {
                ModLogger.info(LOG, "Facehugger " + BoundaryServices.presentation().displayName(facehugger)
                        + " starting attachment business logic to " + BoundaryServices.presentation().displayName(host));
                if (attemptAttachment(facehugger, host)) {
                    setState(facehugger, AttachmentState.ATTACHING);
                    ModLogger.info(LOG, "Facehugger " + BoundaryServices.presentation().displayName(facehugger)
                            + " successfully started attachment to " + BoundaryServices.presentation().displayName(host));
                    yield true;
                }
                ModLogger.warn(LOG, "Facehugger " + BoundaryServices.presentation().displayName(facehugger)
                        + " failed to start attachment to " + BoundaryServices.presentation().displayName(host));
                setState(facehugger, AttachmentState.FREE);
                yield false;
            }
            case ATTACHING, ATTACHED, DETACHING, SPENT_WANDERING -> true;
            default -> true;
        };
    }

    private static void applyAttachmentEffects(@Nonnull LivingEntityRef facehugger, @Nonnull LivingEntityRef host) {
        int fullDuration = AttachmentConfigHelper.getAttachmentDuration()
                + AttachmentConfigHelper.getImplantationDelay();
        EffectApplyOptions options = new EffectApplyOptions(false, false, true);
        BoundaryServices.effects().apply(host, DomainEffectKey.FACEHUGGER_ATTACHMENT, fullDuration, 0, options);
        BoundaryServices.effects().apply(facehugger, DomainEffectKey.FACEHUGGER_IMPLANTATION, fullDuration, 0,
                new EffectApplyOptions(false, false, false));
    }

    private static void removeAttachmentEffect(LivingEntityRef host) {
        BoundaryServices.effects().remove(host, DomainEffectKey.FACEHUGGER_ATTACHMENT);
    }

    private static void removeImplantationEffect(LivingEntityRef facehugger) {
        BoundaryServices.effects().remove(facehugger, DomainEffectKey.FACEHUGGER_IMPLANTATION);
    }

    private static boolean supportsAttachment(LivingEntityRef facehugger) {
        return AttachmentServices.isInitialized() && AttachmentServices.attachment().supports(facehugger);
    }

    private static AttachmentStatePort port() {
        return AttachmentServices.attachment();
    }

    private static AttachmentState getState(LivingEntityRef facehugger) {
        return port().getState(facehugger);
    }

    private static void setState(@Nonnull LivingEntityRef facehugger, @Nonnull AttachmentState state) {
        port().setState(facehugger, state);
    }

    private static void setTicks(@Nonnull LivingEntityRef facehugger, int ticks) {
        port().setTicks(facehugger, ticks);
    }

    private static void setTarget(@Nonnull LivingEntityRef facehugger, @Nullable LivingEntityRef host) {
        port().setTarget(facehugger, host);
    }

    private static boolean isSpent(LivingEntityRef facehugger) {
        return getState(facehugger) == AttachmentState.SPENT_WANDERING;
    }

    private static float getHostYaw(LivingEntityRef host) {
        return BoundaryServices.presentation().spawnContextAt(host).yRot();
    }
}

import com.weylandyutani.layer_0_domain.level_1.entity.LivingEntityRef;

import java.util.HashMap;
import java.util.Map;

/**
 * Central transformation system keyed by {@link XenomorphType}.
 *
 * FILE STATUS: FLUX
 */
public class TransformationSystem {

    private static final Map<XenomorphType, Map<String, TransformationHandler>> REGISTRY = new HashMap<>();

    static {
        registerTransformations();
    }

    private static void registerTransformations() {
        Map<String, TransformationHandler> chestbursterTransforms = new HashMap<>();
        chestbursterTransforms.put("drone", entity -> transformTo(entity, XenomorphType.DRONE));
        REGISTRY.put(XenomorphType.CHESTBURSTER, chestbursterTransforms);

        Map<String, TransformationHandler> droneTransforms = new HashMap<>();
        droneTransforms.put("queen", entity -> transformTo(entity, XenomorphType.QUEEN));
        REGISTRY.put(XenomorphType.DRONE, droneTransforms);
    }

    private static boolean transformTo(LivingEntityRef entity, XenomorphType targetType) {
        if (!BoundaryServices.isInitialized()) {
            return false;
        }
        if (BoundaryServices.entityQuery().isClientSide(entity)) {
            return false;
        }
        return BoundaryServices.entitySpawn().transformXenomorph(entity, targetType) != null;
    }

    public static boolean transformEntity(LivingEntityRef entity, String targetStage) {
        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery()) || targetStage == null || !BoundaryServices.isInitialized()) {
            return false;
        }

        XenomorphType sourceType = BoundaryServices.formQuery().resolveForm(entity);
        if (sourceType == null) {
            ModLogger.warn("TransformationSystem", "No transformation handler for unknown entity");
            return false;
        }

        Map<String, TransformationHandler> entityTransforms = REGISTRY.get(sourceType);
        if (entityTransforms == null) {
            ModLogger.warn("TransformationSystem", "No transformation handler found for type: " + sourceType);
            return false;
        }

        TransformationHandler handler = entityTransforms.get(targetStage.toLowerCase());
        if (handler == null) {
            ModLogger.warn("TransformationSystem", "Unknown transformation target for " + sourceType + ": " + targetStage);
            return false;
        }

        return handler.transform(entity);
    }

    public static boolean canTransform(LivingEntityRef entity, String targetStage) {
        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery()) || targetStage == null || !BoundaryServices.isInitialized()) {
            return false;
        }

        XenomorphType sourceType = BoundaryServices.formQuery().resolveForm(entity);
        if (sourceType == null) {
            return false;
        }

        Map<String, TransformationHandler> entityTransforms = REGISTRY.get(sourceType);
        return entityTransforms != null && entityTransforms.containsKey(targetStage.toLowerCase());
    }

    public static String getTransformationRequirements(LivingEntityRef entity, String targetStage) {
        if (targetStage == null || entity == null || !BoundaryServices.isInitialized()) {
            return "Invalid entity or target stage";
        }

        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery())) {
            return "Invalid entity or target stage";
        }

        if (!canTransform(entity, targetStage)) {
            XenomorphType sourceType = BoundaryServices.isInitialized()
                    ? BoundaryServices.formQuery().resolveForm(entity)
                    : null;
            return "Transformation not available for " + sourceType + " to " + targetStage;
        }

        EntityConfig config = GrowthEntityTypeHelper.getGrowthConfig(entity);
        if (config != null) {
            XenomorphType sourceType = BoundaryServices.formQuery().resolveForm(entity);
            return String.format("%s must be fully matured (%d ticks) to transform to %s",
                    sourceType, config.maturationTime, targetStage);
        }

        XenomorphType sourceType = BoundaryServices.formQuery().resolveForm(entity);
        return String.format("%s must meet maturation requirements to transform to %s", sourceType, targetStage);
    }
}

    public static boolean processGrowth(@Nonnull LivingEntityRef entity) {
        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery())) {
            return false;
        }

        MaturationSystem.processMaturation(entity);

        if (canTransform(entity)) {
            String targetStage = getNextStage(entity);
            if (targetStage != null) {
                return transformEntity(entity, targetStage);
            }
        }

        return false;
    }

    public static boolean canTransform(@Nonnull LivingEntityRef entity) {
        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery())) {
            return false;
        }

        EntityConfig config = GrowthEntityTypeHelper.getGrowthConfig(entity);
        if (config == null) {
            return false;
        }

        int timer = MaturationSystem.getMaturationTimer(entity);
        return timer >= config.maturationTime;
    }

    public static boolean transformEntity(@Nonnull LivingEntityRef entity, String targetStage) {
        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery()) || targetStage == null || !BoundaryServices.isInitialized()) {
            return false;
        }

        ModLogger.info("EntityGrowthSystem", "Transforming "
                + BoundaryServices.presentation().displayName(entity)
                + " to stage: " + targetStage);

        return TransformationSystem.transformEntity(entity, targetStage);
    }

    public static GrowthStage getGrowthStage(LivingEntityRef entity) {
        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery())) {
            return null;
        }

        XenomorphType xenomorphType = XenomorphTypeHelper.getXenomorphType(entity);
        if (xenomorphType == null) {
            return null;
        }

        return switch (xenomorphType) {
            case CHESTBURSTER -> GrowthStage.CHESTBURSTER;
            case DRONE -> GrowthStage.DRONE;
            default -> null;
        };
    }

    public static double getGrowthProgress(@Nonnull LivingEntityRef entity) {
        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery())) {
            return 0.0;
        }

        EntityConfig config = GrowthEntityTypeHelper.getGrowthConfig(entity);
        if (config == null || config.maturationTime <= 0) {
            return 0.0;
        }

        int timer = MaturationSystem.getMaturationTimer(entity);
        return Math.min(1.0, (double) timer / config.maturationTime);
    }

    public static String getNextStage(@Nonnull LivingEntityRef entity) {
        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery())) {
            return null;
        }

        EntityConfig config = GrowthEntityTypeHelper.getGrowthConfig(entity);
        if (config == null || config.matureTo == null || config.matureTo.isEmpty()) {
            return null;
        }

        return config.matureTo;
    }

    public static int getMaturationTimer(@Nonnull LivingEntityRef entity) {
        return MaturationSystem.getMaturationTimer(entity);
    }

    public static void setMaturationTimer(@Nonnull LivingEntityRef entity, int timer) {
        MaturationSystem.setMaturationTimer(entity, timer);
    }

    public static void reduceMaturationTimer(@Nonnull LivingEntityRef entity, int reduction) {
        MaturationSystem.reduceMaturationTimer(entity, reduction);
    }

    public static boolean canGrow(LivingEntityRef entity) {
        if (EntityValidationHelper.isInvalid(entity, BoundaryServices.entityQuery())) {
            return false;
        }
        return GrowthEntityTypeHelper.canMature(entity);
    }

/**
 * Orchestrates facehugger attachment ticks, spent transition, and goal-cancel resets.
 */
public final class FacehuggerTickWorkflow {

    public static final class SpentCallbacks {
        private final Runnable die;
        private final Runnable discard;

        public SpentCallbacks(@Nonnull Runnable die, @Nonnull Runnable discard) {
            this.die = die;
            this.discard = discard;
        }
    }

    private FacehuggerTickWorkflow() {
    }

    /**
     * Domain field reset when implantation completes and the facehugger becomes spent.
     * Caller applies MC side effects (goal rebuild, logging).
     */
    public static void applyMarkAsSpent(@Nonnull FacehuggerAttachmentEntity facehugger) {
        facehugger.setSpentTimer(0);
        facehugger.setAttachmentTarget(null);
        facehugger.setAttachmentTicks(0);
        facehugger.setAttachmentState(AttachmentState.SPENT_WANDERING);
        facehugger.setAttachmentRotationSet(false);
        facehugger.setPendingTargetUUID(null);
    }

    /**
     * Domain field reset when the host dies during attachment.
     * Caller applies MC death side effects.
     */
    public static void applyHostDeathReset(@Nonnull FacehuggerAttachmentEntity facehugger) {
        facehugger.setAttachmentTarget(null);
        facehugger.setAttachmentTicks(0);
        facehugger.setAttachmentState(AttachmentState.FREE);
    }

    /**
     * Domain field reset when the attachment goal is cancelled / stopped.
     */
    public static void applyAttachmentGoalCancelled(@Nonnull FacehuggerAttachmentEntity facehugger) {
        facehugger.setAttachmentTarget(null);
        facehugger.setAttachmentTicks(0);
        facehugger.setAttachmentState(AttachmentRules.onAttachmentGoalCancelled());
    }

    public static int processSpentState(@Nonnull LivingEntityRef facehugger, int spentTimer,
                                        @Nonnull SpentCallbacks callbacks) {
        if (!AttachmentRules.isSpentState(getAttachmentState(facehugger))) {
            return -1;
        }

        int nextTimer = spentTimer + 1;

        int spentWanderTime = AttachmentConfigHelper.getSpentWanderTime();
        if (AttachmentRules.shouldDespawnAfterSpentWander(nextTimer, spentWanderTime)) {
            callbacks.die.run();
            callbacks.discard.run();
        }
        return nextTimer;
    }

    public static void processAttachmentTick(@Nonnull LivingEntityRef facehugger, @Nonnull LivingEntityRef target) {
        if (LivingEntityChecks.isInvalid(target, BoundaryServices.entityQuery())) {
            return;
        }

        AttachmentState attachmentState = getAttachmentState(facehugger);

        if (attachmentState == AttachmentState.READY_TO_ATTACH) {
            FacehuggerAttachmentManager.handleAttachmentStateBusinessLogic(facehugger, target);
        }

        attachmentState = getAttachmentState(facehugger);
        if (!AttachmentRules.shouldProcessImplantationTick(attachmentState)) {
            return;
        }

        boolean shouldContinue = ParasiteLifecycleBootstrap.processAttachmentTick(
                facehugger, target, getAttachmentTicks(facehugger));

        if (shouldContinue) {
            setAttachmentTicks(facehugger, getAttachmentTicks(facehugger) + 1);
        } else if (!AttachmentRules.isSpentState(getAttachmentState(facehugger))) {
            resetAttachmentState(facehugger);
        }
    }

    public static boolean shouldDieWithHost(@Nonnull AttachmentState state, boolean hostAlive) {
        return AttachmentRules.shouldFacehuggerDieWithHost(state, hostAlive);
    }

    private static AttachmentState getAttachmentState(@Nonnull LivingEntityRef facehugger) {
        AttachmentStatePort port = activePort(facehugger);
        return port != null ? port.getState(facehugger) : AttachmentState.FREE;
    }

    private static int getAttachmentTicks(@Nonnull LivingEntityRef facehugger) {
        AttachmentStatePort port = activePort(facehugger);
        return port != null ? port.getTicks(facehugger) : 0;
    }

    private static void setAttachmentTicks(@Nonnull LivingEntityRef facehugger, int ticks) {
        AttachmentStatePort port = activePort(facehugger);
        if (port != null) {
            port.setTicks(facehugger, ticks);
        }
    }

    private static void resetAttachmentState(@Nonnull LivingEntityRef facehugger) {
        AttachmentStatePort port = activePort(facehugger);
        if (port != null) {
            port.setTarget(facehugger, null);
            port.setTicks(facehugger, 0);
            port.setState(facehugger, AttachmentState.FREE);
        }
    }

    @Nullable
    private static AttachmentStatePort activePort(@Nonnull LivingEntityRef facehugger) {
        if (!AttachmentServices.isInitialized()) {
            return null;
        }
        AttachmentStatePort port = AttachmentServices.attachment();
        return port.supports(facehugger) ? port : null;
    }
}

        if (EntityValidationHelper.isInvalid(host, BoundaryServices.entityQuery())) {
            return 0;
        }
        return GestationManager.getRemainingGestationTime(host);
    }
