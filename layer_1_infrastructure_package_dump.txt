package com.weylandyutani.layer_1_infrastructure.level_0.ai;

/**
 * Survival AI behavior parameters loaded from JSON (survival_ai.json).
 * Contains gameplay tuning values for xenomorph survival behavior.
 * 
 * Mirrors JSON structure for `/data/weylandyutani/ai/survival_ai.json`.
 *
 * FILE STATUS: NEW
 *
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class SurvivalAIConfig {
    
    /**
     * Interval between escape attempts (in ticks).
     * Default: 5 ticks (~0.25 seconds at 20 TPS)
     */
    public Integer escapeAttemptIntervalTicks = 5;
    
    /**
     * Maximum time entity can be stuck before giving up (in ticks).
     * Default: 20 ticks (1 second at 20 TPS)
     */
    public Integer stuckThresholdTicks = 20;
    
    /**
     * Low air supply threshold (ratio of current/max).
     * When stuck and air falls below this, entity attempts escape.
     * Default: 0.3 (30% air remaining)
     * Range: 0.0 - 1.0
     */
    public Double suffocationLowThreshold = 0.3;
    
    /**
     * Critical air supply threshold (ratio of current/max).
     * When air falls below this, entity immediately attempts escape.
     * Default: 0.1 (10% air remaining)
     * Range: 0.0 - 1.0
     */
    public Double suffocationCriticalThreshold = 0.1;
    
    public SurvivalAIConfig() {
    }
    
    /**
     * Validate that all values are within acceptable ranges.
     *
     * @throws IllegalArgumentException if any value is invalid
     */
    public void validate() {
        if (escapeAttemptIntervalTicks == null || escapeAttemptIntervalTicks < 1) {
            throw new IllegalArgumentException("escapeAttemptIntervalTicks must be >= 1");
        }
        if (stuckThresholdTicks == null || stuckThresholdTicks < 1) {
            throw new IllegalArgumentException("stuckThresholdTicks must be >= 1");
        }
        if (suffocationLowThreshold == null || suffocationLowThreshold < 0.0 || suffocationLowThreshold > 1.0) {
            throw new IllegalArgumentException("suffocationLowThreshold must be between 0.0 and 1.0");
        }
        if (suffocationCriticalThreshold == null || suffocationCriticalThreshold < 0.0 || suffocationCriticalThreshold > 1.0) {
            throw new IllegalArgumentException("suffocationCriticalThreshold must be between 0.0 and 1.0");
        }
        if (suffocationCriticalThreshold > suffocationLowThreshold) {
            throw new IllegalArgumentException("suffocationCriticalThreshold must be <= suffocationLowThreshold");
        }
    }
}

package com.weylandyutani.layer_1_infrastructure.level_0.block;

import java.util.HashSet;
import java.util.Set;

/**
 * Block rule configuration DTO.
 * Loads from JSON at {@code /data/weylandyutani/block_rules.json}.
 * Defines which blocks cannot be broken by xenomorphs.
 *
 * FILE STATUS: NEW
 *
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class BlockRulesConfig {
    
    /**
     * Set of block registry IDs that cannot be broken.
     * Examples: "minecraft:bedrock", "minecraft:barrier", "minecraft:command_block"
     */
    public Set<String> unbreakable = new HashSet<>();
    
    /**
     * Set of block registry IDs that xenomorphs CAN destroy (whitelist approach).
     * Null or empty means use pattern matching (default behavior).
     * Examples: "minecraft:glass", "minecraft:oak_door"
     */
    public Set<String> breakable = new HashSet<>();
    
    public BlockRulesConfig() {
    }
    
    /**
     * Check if a block is unbreakable.
     *
     * @param blockId registry ID (e.g., "minecraft:stone")
     * @return true if this block cannot be broken
     */
    public boolean isUnbreakable(String blockId) {
        if (blockId == null || blockId.isBlank()) {
            return false;
        }
        return unbreakable.contains(blockId);
    }
    
    /**
     * Check if a block is explicitly breakable.
     *
     * @param blockId registry ID
     * @return true if this block is in the breakable whitelist
     */
    public boolean isExplicitlyBreakable(String blockId) {
        if (blockId == null || blockId.isBlank() || breakable.isEmpty()) {
            return false;
        }
        return breakable.contains(blockId);
    }
}

package com.weylandyutani.layer_1_infrastructure.level_0.effects;

import java.util.EnumMap;
import java.util.Map;

import com.weylandyutani.layer_0_domain.level_0.parasite.HostType;

/**
 * Storage for parsed host effect configurations loaded from classpath JSON.
 */
public final class HostEffectConfigStorage {

    private static final Map<HostType, LoadedHostEffectTypeConfig> HOST_TYPE_CONFIGS = new EnumMap<>(HostType.class);
    private static boolean loaded = false;

    private HostEffectConfigStorage() {
    }

    public static void storeConfig(HostType hostType, LoadedHostEffectTypeConfig config) {
        HOST_TYPE_CONFIGS.put(hostType, config);
    }

    public static LoadedHostEffectTypeConfig getConfig(HostType hostType) {
        if (!loaded) {
            throw new RuntimeException("Host effect configs not loaded. Call HostEffectConfigLoader.loadConfigs() first.");
        }
        LoadedHostEffectTypeConfig config = HOST_TYPE_CONFIGS.get(hostType);
        if (config == null) {
            throw new RuntimeException("No host effect config for HostType: " + hostType);
        }
        return config;
    }

    public static Map<HostType, LoadedHostEffectTypeConfig> getAllConfigs() {
        if (!loaded) {
            throw new RuntimeException("Host effect configs not loaded. Call HostEffectConfigLoader.loadConfigs() first.");
        }
        return Map.copyOf(HOST_TYPE_CONFIGS);
    }

    public static void markLoaded() {
        loaded = true;
    }

    public static boolean isLoaded() {
        return loaded;
    }

    public static void clear() {
        HOST_TYPE_CONFIGS.clear();
        loaded = false;
    }
}

package com.weylandyutani.layer_1_infrastructure.level_0.entity;

import com.weylandyutani.layer_0_domain.level_1.entity.XenomorphType;

/**
 * Resolves AI tuning values from loaded entity JSON via {@link EntityConfigStorage}.
 */
public final class AIConfigResolver {
    private static final int DEFAULT_LIGHT_LEVEL_DARK = 7;
    private static final float DEFAULT_HEALTH_THRESHOLD_FOR_FOOD = 0.75f;
    private static final double DEFAULT_LOW_AIR_SUPPLY_THRESHOLD = 0.3;
    private static final double DEFAULT_CRITICAL_AIR_SUPPLY_THRESHOLD = 0.15;

    private AIConfigResolver() {
    }

    public static double getLowAirSupplyThreshold(XenomorphType type) {
        try {
            EntityConfig config = EntityConfigStorage.getConfig(type);
            if (config != null && config.aiGoals != null && config.aiGoals.lowAirSupplyThreshold != null) {
                return config.aiGoals.lowAirSupplyThreshold;
            }
        } catch (RuntimeException ignored) {
            // Config not loaded yet — use default.
        }
        return DEFAULT_LOW_AIR_SUPPLY_THRESHOLD;
    }

    public static double getCriticalAirSupplyThreshold(XenomorphType type) {
        try {
            EntityConfig config = EntityConfigStorage.getConfig(type);
            if (config != null && config.aiGoals != null && config.aiGoals.criticalAirSupplyThreshold != null) {
                return config.aiGoals.criticalAirSupplyThreshold;
            }
        } catch (RuntimeException ignored) {
            // Config not loaded yet — use default.
        }
        return DEFAULT_CRITICAL_AIR_SUPPLY_THRESHOLD;
    }

    public static int getLightLevelDark(XenomorphType type) {
        try {
            EntityConfig config = EntityConfigStorage.getConfig(type);
            if (config.aiGoals != null && config.aiGoals.movement != null) {
                Object value = config.aiGoals.movement.get("light_level_dark");
                if (value instanceof Number number) {
                    return number.intValue();
                }
            }
        } catch (RuntimeException ignored) {
            // Config not loaded yet or missing movement section — use default.
        }
        return DEFAULT_LIGHT_LEVEL_DARK;
    }

    public static float getHealthThresholdForFood(XenomorphType type) {
        try {
            EntityConfig config = EntityConfigStorage.getConfig(type);
            if (config.aiGoals != null && config.aiGoals.healthThresholdForFood != null) {
                return config.aiGoals.healthThresholdForFood.floatValue();
            }
        } catch (RuntimeException ignored) {
            // Config not loaded yet — use default.
        }
        return DEFAULT_HEALTH_THRESHOLD_FOR_FOOD;
    }
}

package com.weylandyutani.layer_1_infrastructure.level_0.entity;

import com.weylandyutani.layer_0_domain.level_0.ai.PreferredFoodRules;
import com.weylandyutani.layer_0_domain.level_1.entity.XenomorphType;

import java.util.List;

/**
 * L1 config accessors for AI goal parameters (no Minecraft types).
 */
public final class AIGoalConfigHelper {

    private AIGoalConfigHelper() {}

    public static int getMaturationTime(XenomorphType xenomorphType) {
        EntityConfig config = EntityConfigStorage.getConfig(xenomorphType);
        return config.maturationTime;
    }

    public static int getAttachmentDuration(XenomorphType xenomorphType) {
        EntityConfig config = EntityConfigStorage.getConfig(xenomorphType);
        return config.attachmentDuration;
    }

    public static boolean isPreferredFood(String entityTypeKey, EntityConfig.AIGoalsConfig aiGoals) {
        if (aiGoals == null || aiGoals.specialAbilities == null) {
            return false;
        }
        Object preferredFoodObj = aiGoals.specialAbilities.get("preferred_food");
        if (preferredFoodObj == null) {
            return false;
        }

        List<String> preferredFood;
        if (preferredFoodObj instanceof List<?> list) {
            preferredFood = list.stream().filter(String.class::isInstance).map(String.class::cast).toList();
        } else if (preferredFoodObj instanceof String[] array) {
            preferredFood = List.of(array);
        } else {
            return false;
        }

        return PreferredFoodRules.matches(entityTypeKey, preferredFood);
    }
}

package com.weylandyutani.layer_1_infrastructure.level_0.entity;

import java.util.HashMap;
import java.util.Map;

import com.weylandyutani.layer_0_domain.level_1.entity.XenomorphType;

/**
 * Storage for entity configuration objects.
 * Single responsibility: Store and retrieve entity configurations.
 * 
 * FILE STATUS: STABLE
 * 
 * This class handles all storage and retrieval logic for entity configurations.
 * Separated from EntityConfigLoader to follow SRP.
 * 
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class EntityConfigStorage {
    
    private static final Map<XenomorphType, EntityConfig> ENTITY_CONFIGS = new HashMap<>();
    private static final Map<String, EntityConfig.SpeciesVariantConfig> SPECIES_VARIANTS = new HashMap<>();
    private static boolean loaded = false;
    
    /**
     * Stores an entity configuration for a xenomorph type.
     * 
     * @param type The xenomorph type
     * @param config The entity configuration
     */
    public static void storeConfig(XenomorphType type, EntityConfig config) {
        ENTITY_CONFIGS.put(type, config);
    }
    
    /**
     * Gets the configuration for a specific xenomorph type.
     * 
     * @param type The xenomorph type
     * @return The entity configuration
     * @throws RuntimeException if configs not loaded or config not found
     */
    public static EntityConfig getConfig(XenomorphType type) {
        if (!loaded) {
            throw new RuntimeException("Entity configs not loaded yet. Call EntityConfigLoader.loadConfigurations() first.");
        }
        
        EntityConfig config = ENTITY_CONFIGS.get(type);
        if (config == null) {
            throw new RuntimeException("No config found for entity type: " + type + ". Check JSON files.");
        }
        return config;
    }
    
    /**
     * Gets all loaded configurations.
     * 
     * @return Map of all entity configurations
     */
    public static Map<XenomorphType, EntityConfig> getAllConfigs() {
        return new HashMap<>(ENTITY_CONFIGS);
    }
    
    /**
     * Checks if a configuration exists for a xenomorph type.
     * 
     * @param type The xenomorph type
     * @return true if configuration exists
     */
    public static boolean hasConfig(XenomorphType type) {
        return ENTITY_CONFIGS.containsKey(type);
    }
    
    /**
     * Stores a species variant configuration.
     * 
     * @param variantId The variant ID
     * @param config The species variant configuration
     */
    public static void storeSpeciesVariant(String variantId, EntityConfig.SpeciesVariantConfig config) {
        SPECIES_VARIANTS.put(variantId, config);
    }
    
    /**
     * Gets a species variant configuration by variant ID.
     * 
     * @param variantId The variant ID (e.g., "spider", "creeper", "enderman")
     * @return The species variant configuration, or null if not found
     */
    public static EntityConfig.SpeciesVariantConfig getSpeciesVariant(String variantId) {
        return SPECIES_VARIANTS.get(variantId);
    }
    
    /**
     * Gets all loaded species variants.
     * 
     * @return Map of all species variant configurations
     */
    public static Map<String, EntityConfig.SpeciesVariantConfig> getAllSpeciesVariants() {
        return new HashMap<>(SPECIES_VARIANTS);
    }
    
    /**
     * Marks configurations as loaded.
     */
    public static void setLoaded(boolean loaded) {
        EntityConfigStorage.loaded = loaded;
    }
    
    /**
     * Checks if configurations are loaded.
     * 
     * @return true if configurations are loaded
     */
    public static boolean isLoaded() {
        return loaded;
    }
    
    /**
     * Clears all stored configurations.
     * Used when reloading configurations.
     */
    public static void clear() {
        ENTITY_CONFIGS.clear();
        SPECIES_VARIANTS.clear();
    }
    
    /**
     * Gets the number of stored entity configurations.
     * 
     * @return The number of configurations
     */
    public static int getConfigCount() {
        return ENTITY_CONFIGS.size();
    }
    
    /**
     * Gets the number of stored species variant configurations.
     * 
     * @return The number of variants
     */
    public static int getVariantCount() {
        return SPECIES_VARIANTS.size();
    }

     * Gets the shadow size for a xenomorph type.
     * 
     * @param type The xenomorph type
     * @return The shadow size (default 0.5f)

    public static float getShadowSize(XenomorphType type) {

            return (config != null && config.shadowSize > 0) ? config.shadowSize : 0.5f;

        }
    }

     * @param resourcePath The resource path to check

     * @param key The key to retrieve
     * @param defaultValue The default value if key is missing (null for required)
     * @return The boolean value or default
     * @throws RuntimeException if key is missing and defaultValue is null (required parameter)
     */
    public static boolean getBoolean(@Nonnull JsonObject data, @Nonnull String key, @Nullable Boolean defaultValue) {
        if (data.has(key) && data.get(key).isJsonPrimitive()) {
            return data.get(key).getAsBoolean();
        }
        if (defaultValue != null) {
            return defaultValue;
        }
        throw new RuntimeException("Required parameter '" + key + "' missing from JSON config");
    }
    
    /**
     * Gets a required boolean value from JSON object.
     * Throws exception if key is missing.
     * 
     * @param data The JSON object
     * @param key The key to retrieve
     * @return The boolean value
     * @throws RuntimeException if key is missing
     */
    public static boolean getBoolean(@Nonnull JsonObject data, @Nonnull String key) {
        return getBoolean(data, key, null);
    }
    
    /**
     * Gets a list of strings from JSON object.
     * Returns empty list if key is missing or not an array.
     * 
     * @param data The JSON object
     * @param key The key to retrieve
     * @return The list of strings (never null, empty if missing)
     */
    @Nonnull
    public static List<String> getStringList(@Nonnull JsonObject data, @Nonnull String key) {
        if (data.has(key) && data.get(key).isJsonArray()) {
            List<String> strings = new ArrayList<>();
            for (var element : data.getAsJsonArray(key).asList()) {
                strings.add(element.getAsString());
            }
            return strings;
        }
        return new ArrayList<>();
    }
    
    /**
     * Gets a JsonObject from JSON object.
     * 
     * @param data The JSON object
     * @param key The key to retrieve
     * @return The JsonObject, or null if key is missing or not an object
     */
    @Nullable
    public static JsonObject getJsonObject(@Nonnull JsonObject data, @Nonnull String key) {
        if (data.has(key) && data.get(key).isJsonObject()) {
            return data.getAsJsonObject(key);
        }
        return null;
    }
    
    /**
     * Checks if a key exists and is not null in JSON object.
     * 
     * @param data The JSON object
     * @param key The key to check
     * @return True if key exists and is not null
     */
    public static boolean hasKey(@Nonnull JsonObject data, @Nonnull String key) {
        return data.has(key) && !data.get(key).isJsonNull();
    }

    /**
     * Converts a JsonObject into a mutable {@code Map<String,Object>} of primitive values.
     *
     * @param source JSON object
     * @param includeArrays when true, string arrays become {@code List<String>}
     * @param recurseObjects when true, nested objects are converted recursively
     * @return map of converted values (never null)
     */
    @Nonnull
    public static Map<String, Object> toObjectMap(
            @Nonnull JsonObject source, boolean includeArrays, boolean recurseObjects) {
        Map<String, Object> result = new HashMap<>();
        putObjectMap(result, source, includeArrays, recurseObjects);
        return result;
    }

    /**
     * Puts converted JsonObject entries into an existing map (same rules as {@link #toObjectMap}).
     */
    public static void putObjectMap(
            @Nonnull Map<String, Object> target,
            @Nonnull JsonObject source,
            boolean includeArrays,
            boolean recurseObjects) {
        for (String key : source.keySet()) {
            JsonElement value = source.get(key);
            if (value.isJsonPrimitive()) {
                if (value.getAsJsonPrimitive().isBoolean()) {
                    target.put(key, value.getAsBoolean());
                } else if (value.getAsJsonPrimitive().isNumber()) {
                    target.put(key, value.getAsNumber());
                } else {
                    target.put(key, value.getAsString());
                }
            } else if (includeArrays && value.isJsonArray()) {
                List<String> list = new ArrayList<>();
                for (JsonElement item : value.getAsJsonArray()) {
                    list.add(item.getAsString());
                }
                target.put(key, list);
            } else if (recurseObjects && value.isJsonObject()) {
                target.put(key, toObjectMap(Objects.requireNonNull(value.getAsJsonObject()), includeArrays, true));
            }
        }
    }
}

public record LifecycleSettings(

        int defaultTimerReduction,
        int defaultSpentWanderTicks,

        Map<String, Integer> maturation = new HashMap<>();
        maturation.put("ovomorph", 1200);
        maturation.put("chestburster", 4800);
        maturation.put("growth", 6000);
        return new LifecycleSettings(true, true, true, 6000, 600, 200, 100, 100, 40, 20, 4, 100, maturation);
    }

    public static LifecycleSettings parse(JsonObject json) {
        LifecycleSettings defaults = fromJavaDefaults();
        JsonObject defaultsJson = json.has("defaults") ? json.getAsJsonObject("defaults") : null;
        JsonObject attachmentJson = json.has("attachment") ? json.getAsJsonObject("attachment") : null;
        JsonObject maturationJson = json.has("maturation") ? json.getAsJsonObject("maturation") : null;
        JsonObject maturationTimesJson = json.has("maturation_times") ? json.getAsJsonObject("maturation_times") : null;
        Map<String, Integer> maturationTimes = new HashMap<>(defaults.maturationTimesByForm());
        if (maturationTimesJson != null) {
            for (String key : maturationTimesJson.keySet()) {
                maturationTimes.put(key, maturationTimesJson.get(key).getAsInt());
            }
        }
        return new LifecycleSettings(
                boolOr(json, "enabled", defaults.enabled()),
                boolOr(json, "parasite_enabled", defaults.parasiteEnabled()),
                boolOr(json, "growth_enabled", defaults.growthEnabled()),
                intOr(defaultsJson, "maturation_time", defaults.defaultMaturationTime()),
                intOr(defaultsJson, "attachment_duration", defaults.defaultAttachmentDuration()),
                intOr(defaultsJson, "implantation_delay", defaults.defaultImplantationDelay()),
                intOr(defaultsJson, "timer_reduction", defaults.defaultTimerReduction()),
                intOr(defaultsJson, "spent_wander_ticks", defaults.defaultSpentWanderTicks()),
                intOr(defaultsJson, "death_animation_ticks", defaults.defaultDeathAnimationTicks()),
                intOr(attachmentJson, "effect_refresh_ticks", defaults.attachmentEffectRefreshTicks()),
                intOr(maturationJson, "chestburster_log_interval_divisor", defaults.chestbursterLogIntervalDivisor()),
                intOr(maturationJson, "drone_log_interval", defaults.droneLogInterval()),
                maturationTimes
        );
    }

    public int maturationTimeForForm(String formKey) {
        if (formKey == null) {
            return defaultMaturationTime;
        }
        return maturationTimesByForm.getOrDefault(formKey.toLowerCase(), defaultMaturationTime);
    }

    private static boolean boolOr(JsonObject json, String key, boolean fallback) {
        return json != null && json.has(key) ? json.get(key).getAsBoolean() : fallback;
    }

    private static int intOr(JsonObject json, String key, int fallback) {
        return json != null && json.has(key) ? json.get(key).getAsInt() : fallback;
    }
}

        while (cursor < pattern.length()) {
            int placeholder = pattern.indexOf("{}", cursor);
            if (placeholder < 0 || argIndex >= args.length) {
                result.append(pattern.substring(cursor));
                break;
            }
            result.append(pattern, cursor, placeholder);
            result.append(args[argIndex++]);
            cursor = placeholder + 2;
        }
        return result.toString();

package com.weylandyutani.layer_1_infrastructure.level_1.block;

import javax.annotation.Nullable;

import com.weylandyutani.layer_1_infrastructure.level_0.block.LoadedBlockTextureConfig;
import com.weylandyutani.layer_1_infrastructure.level_0.block.LoadedBlockVisualConfig;

/**
 * Resolves texture paths from a block visual config and {@link BlockTextureRegistry}.
 */
public final class BlockTextureResolver {

    private static final String FALLBACK = "minecraft:block/black_concrete";

    private BlockTextureResolver() {
    }

    public static String resolveSingle(@Nullable LoadedBlockVisualConfig config) {
        if (config == null || config.textureKey == null) {
            return FALLBACK;
        }
        LoadedBlockTextureConfig textureConfig = BlockTextureRegistry.get(config.textureKey);
        if (textureConfig == null || !"single".equals(textureConfig.type)) {
            return FALLBACK;
        }
        return textureConfig.texture != null ? textureConfig.texture : textureConfig.fallback;
    }

    public static String resolveMulti(@Nullable LoadedBlockVisualConfig config, String side) {
        if (config == null || config.textureKey == null) {
            return FALLBACK;
        }
        LoadedBlockTextureConfig textureConfig = BlockTextureRegistry.get(config.textureKey);
        if (textureConfig == null || !"multi".equals(textureConfig.type)) {
            return FALLBACK;
        }

        String texture = switch (side) {
            case "sides" -> textureConfig.sides;
            case "top" -> textureConfig.top;
            case "bottom" -> textureConfig.bottom;
            case "front" -> textureConfig.front;
            case "back" -> textureConfig.back;
            default -> null;
        };

        if (texture == null && textureConfig.texture != null) {
            texture = textureConfig.texture;
        }
        return texture != null ? texture : textureConfig.fallback;
    }
}

import com.weylandyutani.layer_1_infrastructure.level_0.json.JsonParsingHelper;
import com.weylandyutani.layer_1_infrastructure.level_0.registry.ConfigContractException;

import java.util.ArrayList;
import java.util.List;
import java.util.Objects;
import javax.annotation.Nonnull;

/**
 * Parser for {@code systems/host_effects.json}.
 */
public final class HostEffectConfigParser {

    private static final String RESOURCE = "/data/weylandyutani/systems/host_effects.json";
    private static final String LOADER = "HostEffectConfigLoader";

    private HostEffectConfigParser() {
    }

    public static void parseAndStore(@Nonnull JsonObject hostTypes) {
        HostEffectConfigStorage.clear();
        for (HostType hostType : HostType.values()) {
            if (!hostTypes.has(hostType.getKey()) || !hostTypes.get(hostType.getKey()).isJsonObject()) {
                throw new ConfigContractException(LOADER, RESOURCE, "host_types." + hostType.getKey(), "required");
            }
            JsonObject typeData = Objects.requireNonNull(hostTypes.getAsJsonObject(hostType.getKey()));
            HostEffectConfigStorage.storeConfig(hostType, parseHostTypeConfig(typeData, hostType.getKey()));
        }
        HostEffectConfigStorage.markLoaded();
    }

    private static LoadedHostEffectTypeConfig parseHostTypeConfig(@Nonnull JsonObject typeData, String hostTypeKey) {
        boolean requiresPlayerSpecificEffects = JsonParsingHelper.getBoolean(
                typeData, "requires_player_specific_effects");
        boolean canBeImmobilized = JsonParsingHelper.getBoolean(typeData, "can_be_immobilized");
        String description = JsonParsingHelper.getString(typeData, "description");
        List<LoadedHostEffectEntry> visualEffects = parseEffectList(typeData, "visual_effects", hostTypeKey);
        List<LoadedHostEffectEntry> movementEffects = parseEffectList(typeData, "movement_effects", hostTypeKey);
        return new LoadedHostEffectTypeConfig(
                visualEffects,
                movementEffects,
                requiresPlayerSpecificEffects,
                canBeImmobilized,
                description);
    }

    private static List<LoadedHostEffectEntry> parseEffectList(@Nonnull JsonObject typeData, String key, String hostTypeKey) {
        if (!typeData.has(key) || !typeData.get(key).isJsonArray()) {
            throw new ConfigContractException(LOADER, RESOURCE,
                    "host_types." + hostTypeKey + "." + key, "required array");
        }
        JsonArray effects = typeData.getAsJsonArray(key);
        List<LoadedHostEffectEntry> parsed = new ArrayList<>();
        for (JsonElement element : effects) {
            if (!element.isJsonObject()) {
                throw new ConfigContractException(LOADER, RESOURCE,
                        "host_types." + hostTypeKey + "." + key, "array elements must be objects");
            }
            parsed.add(parseEffectEntry(Objects.requireNonNull(element.getAsJsonObject()), hostTypeKey, key));
        }
        return parsed;
    }

    private static LoadedHostEffectEntry parseEffectEntry(@Nonnull JsonObject effectData, String hostTypeKey, String listKey) {
        String effectKey = JsonParsingHelper.getString(effectData, "effect");
        int duration = JsonParsingHelper.getInt(effectData, "duration");
        int amplifier = JsonParsingHelper.getInt(effectData, "amplifier");
        boolean showParticles = JsonParsingHelper.getBoolean(effectData, "show_particles");
        String description = JsonParsingHelper.getString(effectData, "description");
        try {
            StandardMobEffectKey standardEffect = resolveStandardEffectKey(effectKey);
            return new LoadedHostEffectEntry(standardEffect, duration, amplifier, showParticles, description);
        } catch (ConfigContractException ex) {
            throw new ConfigContractException(LOADER, RESOURCE,
                    "host_types." + hostTypeKey + "." + listKey + ".effect",
                    "unknown effect key '" + effectKey + "'");
        }
    }

    private static StandardMobEffectKey resolveStandardEffectKey(String effectKey) {
        return switch (effectKey) {
            case "blindness" -> StandardMobEffectKey.BLINDNESS;
            case "weakness" -> StandardMobEffectKey.WEAKNESS;
            case "hunger" -> StandardMobEffectKey.HUNGER;
            case "poison" -> StandardMobEffectKey.POISON;
            case "movement_slowdown" -> StandardMobEffectKey.SLOWNESS;
            case "dig_slowdown" -> StandardMobEffectKey.DIG_SLOWDOWN;
            case "slow_falling" -> StandardMobEffectKey.SLOW_FALLING;
            case "levitation" -> StandardMobEffectKey.LEVITATION;
            default -> throw new ConfigContractException(LOADER, RESOURCE, "effect", "unsupported key");
        };
    }
}

import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

/**
 * Helper class for caching entity configuration loads.
 * Centralizes config loading pattern with caching to eliminate DRY violations.
 * 
 * DRY: Eliminates duplicate EntityConfigStorage.getConfig() calls with caching logic across 10+ files.
 * 
 * FILE STATUS: STABLE
 * 
 * @author WeylandYutaniCorp
 * @since 0.1.8
 */
public class ConfigCacheHelper {
    
    // Thread-safe cache for entity configs
    private static final Map<XenomorphType, EntityConfig> configCache = new ConcurrentHashMap<>();
    
    /**
     * Gets entity configuration for the specified xenomorph type.
     * Uses caching to avoid redundant loads.
     * 
     * @param type The xenomorph type
     * @return The entity configuration, or null if not found
     */
    public static EntityConfig getConfig(XenomorphType type) {
        if (type == null) {
            return null;
        }
        
        // Check cache first
        EntityConfig cached = configCache.get(type);
        if (cached != null) {
            return cached;
        }
        
        // Load from storage
        EntityConfig config = EntityConfigStorage.getConfig(type);
        
        // Cache if not null (cache nulls too to avoid repeated lookups)
        if (config != null) {
            configCache.put(type, config);
        }
        
        return config;
    }
    
    /**
     * Clears the configuration cache.
     * Useful for testing or when configs are reloaded.
     */
    public static void clearCache() {
        configCache.clear();
    }
    
    /**
     * Clears cached config for a specific type.
     * Useful when a specific config is reloaded.
     * 
     * @param type The xenomorph type to clear from cache
     */
    public static void clearCache(XenomorphType type) {
        if (type != null) {
            configCache.remove(type);
        }
    }
    
    // Private constructor to prevent instantiation
    private ConfigCacheHelper() {
        throw new UnsupportedOperationException("ConfigCacheHelper is a utility class and cannot be instantiated");
    }

    };

    private static final String[] OVOMORPH_GOAL_PRIORITY_KEYS = {
            "float", "hatch", "random_look", "look"
    };

    private static final String[] CHESTBURSTER_MOVEMENT_KEYS = {
            "flee_distance", "escape_speed_multiplier"
    };

    private static final String[] QUEEN_SPECIAL_ABILITY_KEYS = {
            "hive_mother_speed_multiplier",
            "egg_laying_interval",
            "min_distance_between_eggs",
            "max_eggs_per_chunk",
            "egg_search_radius",

            "stroll_interval",
            "wander_interval",
            "nest_radius",
            "avoidance_distance_multiplier",

    };

    private static final String[] DRONE_SPECIAL_ABILITY_KEYS = {
            "patrol_interval",
            "swarm_stroll_speed_multiplier",
            "swarm_look_range_multiplier"
    };

    private EntityAIGoalsContract() {
    }

    public static void validate(EntityConfig config, XenomorphType type, String resourcePath) {
        if (config.aiGoals == null) {
            throw new ConfigContractException(LOADER, resourcePath, "ai_goals", "required");
        }
        if (config.aiGoals.behaviorType == null || config.aiGoals.behaviorType.isBlank()) {
            throw new ConfigContractException(LOADER, resourcePath, "ai_goals.behavior_type", "required");
        }
        if (!AIBehaviorConfigNames.shippedSet().contains(config.aiGoals.behaviorType)) {
            throw new ConfigContractException(LOADER, resourcePath, "ai_goals.behavior_type",
                    "unknown behavior type: " + config.aiGoals.behaviorType);
        }
        requireMovement(config, type, resourcePath);
        requireTypeSpecialAbilities(config, type, resourcePath);
        requireGoalPriorities(config, type, resourcePath);
    }

    public static List<String> contractedGoalPriorityKeys(XenomorphType type) {
        return switch (type) {
            case DRONE -> combineKeys(DRONE_GOAL_PRIORITY_KEYS, STANDARD_TARGET_PRIORITY_KEYS);
            case QUEEN -> combineKeys(QUEEN_GOAL_PRIORITY_KEYS, STANDARD_TARGET_PRIORITY_KEYS);
            case CHESTBURSTER -> List.of(CHESTBURSTER_GOAL_PRIORITY_KEYS);
            case PRAETOR -> List.of(PRAETOR_GOAL_PRIORITY_KEYS);
            case FACEHUGGER -> List.of(FACEHUGGER_GOAL_PRIORITY_KEYS);
            case OVOMORPH -> List.of(OVOMORPH_GOAL_PRIORITY_KEYS);
            default -> List.of();
        };
    }

    public static Set<String> queenSpecialAbilityKeys() {
        return Set.of(QUEEN_SPECIAL_ABILITY_KEYS);
    }

    private static void requireMovement(EntityConfig config, XenomorphType type, String resourcePath) {
        Map<String, Object> movement = config.aiGoals.movement;
        if (movement == null) {
            throw new ConfigContractException(LOADER, resourcePath, "ai_goals.movement", "required");
        }
        for (String key : MOVEMENT_KEYS) {
            requireNumber(movement, key, "ai_goals.movement." + key, resourcePath);
        }
        if (type == XenomorphType.CHESTBURSTER) {
            for (String key : CHESTBURSTER_MOVEMENT_KEYS) {
                requireNumber(movement, key, "ai_goals.movement." + key, resourcePath);
            }
        }
    }

    private static void requireTypeSpecialAbilities(EntityConfig config, XenomorphType type, String resourcePath) {
        List<String> required = switch (type) {
            case QUEEN -> List.of(QUEEN_SPECIAL_ABILITY_KEYS);
            case DRONE -> List.of(DRONE_SPECIAL_ABILITY_KEYS);
            case CHESTBURSTER -> List.of(
                    "eat_small_animals", "attack_distance", "food_healing", "maturation_timer_reduction");
            case GROWTH -> List.of("passive_stroll_speed_multiplier");
            case OVOMORPH -> List.of("proximity_detection");
            case PRAETOR -> List.of(
                    "strategic_melee_speed_multiplier",
                    "strategic_stroll_speed_multiplier",
                    "command_look_range_multiplier");
            default -> List.of();
        };
        if (required.isEmpty()) {
            return;
        }
        Map<String, Object> abilities = config.aiGoals.specialAbilities;
        if (abilities == null) {
            throw new ConfigContractException(LOADER, resourcePath, "ai_goals.special_abilities", "required for " + type);
        }
        for (String key : required) {
            if (!abilities.containsKey(key)) {
                throw new ConfigContractException(LOADER, resourcePath, "ai_goals.special_abilities." + key, "required");
            }
        }
    }

    private static void requireGoalPriorities(EntityConfig config, XenomorphType type, String resourcePath) {
        List<String> required = contractedGoalPriorityKeys(type);
        if (required.isEmpty()) {
            return;
        }
        if (config.aiGoals.goalPriorities == null) {
            throw new ConfigContractException(LOADER, resourcePath, "ai_goals.goal_priorities", "required for " + type);
        }
        for (String key : required) {
            Integer priority = config.aiGoals.goalPriorities.get(key);
            if (priority == null) {
                throw new ConfigContractException(
                        LOADER, resourcePath, "ai_goals.goal_priorities." + key, "required for " + type);
            }
            if (priority < 0 || priority > 10) {
                throw new ConfigContractException(
                        LOADER, resourcePath, "ai_goals.goal_priorities." + key, "must be between 0 and 10");
            }
        }
    }

    private static List<String> combineKeys(String[] goalKeys, String[] targetKeys) {
        List<String> combined = new java.util.ArrayList<>(List.of(goalKeys));
        combined.addAll(List.of(targetKeys));
        return combined;
    }

    private static void requireNumber(Map<String, Object> map, String key, String fieldPath, String resourcePath) {
        if (!map.containsKey(key)) {
            throw new ConfigContractException(LOADER, resourcePath, fieldPath, "required");
        }
        Object value = map.get(key);
        if (!(value instanceof Number)) {
            throw new ConfigContractException(LOADER, resourcePath, fieldPath, "must be numeric");
        }
    }
}

package com.weylandyutani.layer_1_infrastructure.level_1.entity;

import java.util.function.Function;

/**
 * Centralized utility for entity configuration access.
 * Consolidates duplicate configuration access patterns across entity classes.
 * 
 * FILE STATUS: STABLE
 * 
 * This utility consolidates configuration access patterns following DRY principles.
 * Reduces boilerplate code for null checking and error handling when accessing entity configs.
 * 
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class EntityConfigHelper {
    
    /**
     * Gets a configuration value from entity config with null-safe extraction.
     * 
     * @param <T> The type of the configuration value to extract
     * @param xenomorphType The xenomorph type to get configuration for
     * @param extractor Function to extract the desired value from the config
     * @param defaultValue Default value to return if config or value is null
     * @return The extracted configuration value, or defaultValue if not found
     */
    public static <T> T getConfigValue(
            XenomorphType xenomorphType,
            Function<EntityConfig, T> extractor,
            T defaultValue) {
        
        if (xenomorphType == null || extractor == null) {
            return defaultValue;
        }
        
        try {
            EntityConfig config = EntityConfigStorage.getConfig(xenomorphType);
            if (config == null) {
                return defaultValue;
            }
            
            T value = extractor.apply(config);
            return value != null ? value : defaultValue;
        } catch (Exception e) {
            ModLogger.debug("EntityConfigHelper", 
                String.format("Failed to extract config value for %s: %s", xenomorphType, e.getMessage()));
            return defaultValue;
        }
    }
    
    /**
     * Gets a configuration value from entity config with validation.
     * Throws exception if config is not found or value is invalid.
     * 
     * @param <T> The type of the configuration value to extract
     * @param xenomorphType The xenomorph type to get configuration for
     * @param extractor Function to extract the desired value from the config
     * @param errorMessage Error message to use if extraction fails
     * @return The extracted configuration value
     * @throws RuntimeException If configuration cannot be loaded or value is invalid
     */
    public static <T> T getConfigValueRequired(
            XenomorphType xenomorphType,
            Function<EntityConfig, T> extractor,
            String errorMessage) {
        
        if (xenomorphType == null) {
            throw new IllegalArgumentException("XenomorphType cannot be null");
        }
        
        if (extractor == null) {
            throw new IllegalArgumentException("Extractor function cannot be null");
        }
        
        EntityConfig config = EntityConfigStorage.getConfig(xenomorphType);
        if (config == null) {
            ModLogger.error("EntityConfigHelper", 
                String.format("❌ No config found for %s - %s", xenomorphType, errorMessage));
            throw new RuntimeException(String.format("Entity configuration not found: %s - %s", xenomorphType, errorMessage));
        }
        
        try {
            T value = extractor.apply(config);
            if (value == null) {
                ModLogger.error("EntityConfigHelper", 
                    String.format("❌ %s is null in %s configuration - %s", errorMessage, xenomorphType, errorMessage));
                throw new RuntimeException(String.format("%s is null in %s configuration", errorMessage, xenomorphType));
            }
            return value;
        } catch (Exception e) {
            ModLogger.error("EntityConfigHelper", 
                String.format("❌ Failed to extract %s from %s config: %s", errorMessage, xenomorphType, e.getMessage()));
            throw new RuntimeException(String.format("Failed to extract %s from %s config: %s", errorMessage, xenomorphType, e.getMessage()), e);
        }
    }
}

import com.weylandyutani.layer_1_infrastructure.level_0.registry.ConfigContractException;

/**
 * Contract for entity JSON {@code visual} fields (model + sound resource ids).
 */
public final class EntityVisualContract {

    private static final String LOADER = "EntityConfigLoader";

    private EntityVisualContract() {
    }

    public static void validate(EntityConfig config, String resourcePath) {
        if (config.model == null || config.model.isBlank()) {
            throw new ConfigContractException(LOADER, resourcePath, "visual.model", "required");
        }
        requireSoundField(config.ambientSound, "visual.ambient_sound", resourcePath);
        requireSoundField(config.hurtSound, "visual.hurt_sound", resourcePath);
        requireSoundField(config.deathSound, "visual.death_sound", resourcePath);
    }

    private static void requireSoundField(String value, String fieldPath, String resourcePath) {
        if (value == null || value.isBlank()) {
            throw new ConfigContractException(LOADER, resourcePath, fieldPath, "required");
        }
        if (!isNamespacedResourceId(value)) {
            throw new ConfigContractException(LOADER, resourcePath, fieldPath,
                    "must be a namespaced resource id (namespace:path)");
        }
    }

    private static boolean isNamespacedResourceId(String value) {
        int colon = value.indexOf(':');
        if (colon <= 0 || colon == value.length() - 1) {
            return false;
        }
        String namespace = value.substring(0, colon);
        String path = value.substring(colon + 1);
        return !namespace.isBlank() && !path.isBlank();
    }
}

 * This class handles all JSON parsing logic for host configurations.
 * Separated from HostConfigLoader to follow SRP.
 * 
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class HostConfigParser {
    
    /**
     * Parses JSON data into HostConfig object.
     * 
     * @param jsonData The JSON object to parse
     * @return The parsed HostConfig
     */
    public static HostConfig parseHostConfiguration(@Nonnull JsonObject jsonData) {
        HostConfig config = new HostConfig();
        
        // Parse vanilla hosts
        JsonObject vanillaHosts = JsonParsingHelper.getJsonObject(jsonData, "vanilla_hosts");
        if (vanillaHosts != null) {
            parseHostCategory(vanillaHosts, "vanilla", config);
        }
        
        // Parse modded hosts (organized by mod namespace)
        JsonObject moddedHosts = JsonParsingHelper.getJsonObject(jsonData, "modded_hosts");
        if (moddedHosts != null) {
            // Iterate through each mod namespace (e.g., "weylandyutani", "other_mod", etc.)
            for (String modNamespace : moddedHosts.keySet()) {
                JsonElement modData = moddedHosts.get(modNamespace);
                
                // Each mod namespace contains host categories
                if (modData.isJsonObject()) {
                    JsonObject modHostsConfig = Objects.requireNonNull(modData.getAsJsonObject());
                    parseHostCategory(modHostsConfig, "modded:" + modNamespace, config);
                    ModLogger.debug("HostConfigParser", 
                        String.format("   - Loaded modded hosts from namespace: %s", modNamespace));
                }
            }
        }
        
        // Parse variant mappings
        JsonObject mappings = JsonParsingHelper.getJsonObject(jsonData, "variant_mappings");
        if (mappings != null) {
            for (String key : mappings.keySet()) {
                if (!key.equals("comment")) {  // Skip comment fields
                    JsonElement element = mappings.get(key);
                    if (element.isJsonPrimitive()) {
                        config.variantMappings.put(key.toLowerCase(), element.getAsString());
                    }
                }
            }
        }
        
        return config;
    }
    
    /**
     * Parses a category of hosts (vanilla or modded).
     * 
     * @param category The category JSON object
     * @param categoryName The name of the category (for logging)
     * @param config The HostConfig to populate
     */
    private static void parseHostCategory(@Nonnull JsonObject category, @Nonnull String categoryName, @Nonnull HostConfig config) {
        // Parse living biological hosts
        if (JsonParsingHelper.hasKey(category, "living_biological") && category.get("living_biological").isJsonArray()) {
            JsonArray hosts = category.getAsJsonArray("living_biological");
            for (JsonElement element : hosts) {
                config.hostTypeMap.put(element.getAsString().toLowerCase(), HostType.LIVING_BIOLOGICAL);
            }
            ModLogger.debug("HostConfigParser", 
                String.format("   - %d living_biological hosts loaded from %s", hosts.size(), categoryName));
        }
        
        // Parse undead hosts
        if (JsonParsingHelper.hasKey(category, "undead") && category.get("undead").isJsonArray()) {
            JsonArray hosts = category.getAsJsonArray("undead");
            for (JsonElement element : hosts) {
                config.hostTypeMap.put(element.getAsString().toLowerCase(), HostType.UNDEAD);
            }
            ModLogger.debug("HostConfigParser", 
                String.format("   - %d undead hosts loaded from %s", hosts.size(), categoryName));
        }
        
        // Parse unsuitable hosts
        JsonObject unsuitable = JsonParsingHelper.getJsonObject(category, "unsuitable");
        if (unsuitable != null) {
            // Constructs
            if (JsonParsingHelper.hasKey(unsuitable, "construct") && unsuitable.get("construct").isJsonArray()) {
                JsonArray hosts = unsuitable.getAsJsonArray("construct");
                for (JsonElement element : hosts) {
                    config.hostTypeMap.put(element.getAsString().toLowerCase(), HostType.CONSTRUCT);
                }
            }
            
            // Inorganic
            if (JsonParsingHelper.hasKey(unsuitable, "inorganic") && unsuitable.get("inorganic").isJsonArray()) {
                JsonArray hosts = unsuitable.getAsJsonArray("inorganic");
                for (JsonElement element : hosts) {
                    config.hostTypeMap.put(element.getAsString().toLowerCase(), HostType.INORGANIC);
                }
            }
            
            // Elemental
            if (JsonParsingHelper.hasKey(unsuitable, "elemental") && unsuitable.get("elemental").isJsonArray()) {
                JsonArray hosts = unsuitable.getAsJsonArray("elemental");
                for (JsonElement element : hosts) {
                    config.hostTypeMap.put(element.getAsString().toLowerCase(), HostType.ELEMENTAL);
                }
            }
            
            // Spiritual
            if (JsonParsingHelper.hasKey(unsuitable, "spiritual") && unsuitable.get("spiritual").isJsonArray()) {
                JsonArray hosts = unsuitable.getAsJsonArray("spiritual");
                for (JsonElement element : hosts) {
                    config.hostTypeMap.put(element.getAsString().toLowerCase(), HostType.SPIRITUAL);
                }
            }
        }
    }

/**

 * Used by static config loaders that load from classpath resources.
 * 
 * FILE STATUS: STABLE
 * 
 * This implementation wraps classpath resource loading,
 * providing a ResourceLoader interface for config loading.
 * 
 * @author WeylandYutaniCorp
 * @since 0.1.0

public final class ClasspathResourceLoader implements ResourceLoader {
    
    public static final ClasspathResourceLoader INSTANCE = new ClasspathResourceLoader();

    }

    public JsonObject loadJson(String resourcePath, Class<?> contextClass) {

                ModLogger.warn("ClasspathResourceLoader", "Resource not found: " + resourcePath);
                return null;
            }

            return JsonParser.parseReader(reader).getAsJsonObject();
        } catch (Exception e) {
            ModLogger.error("ClasspathResourceLoader", "Failed to load JSON from " + resourcePath + ": " + e.getMessage(), e);
            return null;
        }
    }
    
    @Override
    public boolean resourceExists(String resourcePath, Class<?> contextClass) {
        return contextClass.getResourceAsStream(resourcePath) != null;
    }

/**
 * Centralized utility for loading JSON resources from classpath.
 * Consolidates duplicate resource loading patterns across config loaders.
 * 
 * FILE STATUS: STABLE
 * 
 * This utility implements DRY principles by centralizing the common pattern:
 * getResourceAsStream() → InputStreamReader → JsonParser.parseReader()
 * 
 * Used by EntityConfigParser, EffectConfigLoader, HostConfigLoader,
 * MobEntityConfigLoader, BlockConfigLoader, and other config loaders.
 * 
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class ResourceLoadingHelper {
    
    /**
     * Loads and parses a JSON file from the classpath into a JsonObject.
     * 
     * @param resourcePath The resource path to the JSON file (e.g., "/data/weylandyutani/effects/gestation.json")
     * @param contextClass The class to use for getResourceAsStream() (usually the calling class)
     * @return The parsed JsonObject, or null if file not found or parsing failed
     */
    public static JsonObject loadJsonFromResource(String resourcePath, Class<?> contextClass) {
        try (InputStream inputStream = contextClass.getResourceAsStream(resourcePath)) {
            if (inputStream == null) {
                ModLogger.debug("ResourceLoadingHelper", 
                    String.format("Resource not found: %s", resourcePath));
                return null;
            }
            
            try (InputStreamReader reader = new InputStreamReader(inputStream, StandardCharsets.UTF_8)) {
                return JsonParser.parseReader(reader).getAsJsonObject();
            }
        } catch (Exception e) {
            ModLogger.error("ResourceLoadingHelper", 
                String.format("Failed to load JSON from %s: %s", resourcePath, e.getMessage()), e);
            return null;
        }
    }
    
    /**
     * Loads and parses a JSON file from the classpath into a JsonObject.
     * Uses the ResourceLoadingHelper class itself for getResourceAsStream().
     * 
     * @param resourcePath The resource path to the JSON file
     * @return The parsed JsonObject, or null if file not found or parsing failed
     */
    public static JsonObject loadJsonFromResource(String resourcePath) {
        return loadJsonFromResource(resourcePath, ResourceLoadingHelper.class);
    }
    
    /**
     * Checks if a resource exists at the given path.
     * 
     * @param resourcePath The resource path to check
     * @param contextClass The class to use for getResourceAsStream()
     * @return True if resource exists, false otherwise
     */
    public static boolean resourceExists(String resourcePath, Class<?> contextClass) {
        try (InputStream inputStream = contextClass.getResourceAsStream(resourcePath)) {
            return inputStream != null;
        } catch (Exception e) {
            return false;
        }
    }
    
    /**
     * Checks if a resource exists at the given path.
     * Uses the ResourceLoadingHelper class itself for getResourceAsStream().
     * 
     * @param resourcePath The resource path to check
     * @return True if resource exists, false otherwise
     */
    public static boolean resourceExists(String resourcePath) {
        return resourceExists(resourcePath, ResourceLoadingHelper.class);
    }
}

        return loaderName;

     * Subclasses can override this for custom validation logic.

     * @param config The configuration to validate
     * @param configName The name of the configuration
     * @return True if the configuration is valid

    protected boolean validateConfig(T config, String configName) {
        return config != null;
    }
    
    @Override
    public boolean validateAllConfigs() {
        boolean allValid = true;
        for (Map.Entry<String, T> entry : configs.entrySet()) {
            if (!validateConfig(entry.getValue(), entry.getKey())) {
                ModLogger.warn("Invalid " + loaderName + " config: " + entry.getKey());
                allValid = false;
            }
        }
        return allValid;
    }
    
    /**
     * Logs configuration access for debugging.
     * 
     * @param configName The name of the configuration being accessed
     */
    protected void logConfigAccess(String configName) {
        ModLogger.debug("Accessed " + loaderName + " config: " + configName);
    }

     * Throws an exception if the configuration is not found or invalid.
     * 
     * @param configName The name of the configuration
     * @return The configuration
     * @throws RuntimeException if configuration is not found or invalid

    protected T getConfigWithValidation(String configName) {
        T config = getConfigInternal(configName);

        }
        if (!validateConfig(config, configName)) {

package com.weylandyutani.layer_1_infrastructure.level_1.registry;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import com.weylandyutani.layer_1_infrastructure.level_0.platform.ModLogger;
import com.weylandyutani.layer_1_infrastructure.level_0.registry.ConfigLoader;

/**
 * Central registry for all configuration loaders.
 * Manages loader priorities and orchestrates loading.
 * 
 * FILE STATUS: STABLE
 */
public final class ConfigLoaderRegistry {
    
    private static final Map<String, ConfigLoader> LOADERS = new HashMap<>();
    private static final Map<String, Integer> PRIORITIES = new HashMap<>();
    
    private ConfigLoaderRegistry() {
    }
    
    public static void registerLoader(String name, ConfigLoader loader) {
        registerLoader(name, loader, 10);
    }
    
    public static void registerLoader(String name, ConfigLoader loader, int priority) {
        LOADERS.put(name, loader);
        PRIORITIES.put(name, priority);
        ModLogger.info("ConfigLoaderRegistry", "Registered loader: " + name + " (priority: " + priority + ")");
    }
    
    public static ConfigLoader getLoader(String name) {
        return LOADERS.get(name);
    }
    
    public static void loadAllConfigs() {
        List<Map.Entry<String, Integer>> sortedLoaders = new ArrayList<>(PRIORITIES.entrySet());
        sortedLoaders.sort((a, b) -> Integer.compare(a.getValue(), b.getValue()));
        
        for (Map.Entry<String, Integer> entry : sortedLoaders) {
            String name = entry.getKey();
            ConfigLoader loader = LOADERS.get(name);
            if (loader != null) {
                try {
                    loader.loadConfigs();
                    ModLogger.info("ConfigLoaderRegistry", "Loaded config: " + name);
                } catch (Exception e) {
                    ModLogger.error("ConfigLoaderRegistry", "Failed to load " + name + ": " + e.getMessage(), e);
                }
            }
        }
    }

    public static void reloadAllConfigs() {
        List<Map.Entry<String, Integer>> sortedLoaders = new ArrayList<>(PRIORITIES.entrySet());
        sortedLoaders.sort((a, b) -> Integer.compare(a.getValue(), b.getValue()));

        for (Map.Entry<String, Integer> entry : sortedLoaders) {
            String name = entry.getKey();
            ConfigLoader loader = LOADERS.get(name);
            if (loader != null) {
                try {
                    loader.reloadConfigs();
                    ModLogger.info("ConfigLoaderRegistry", "Reloaded config: " + name);
                } catch (Exception e) {
                    ModLogger.error("ConfigLoaderRegistry", "Failed to reload " + name + ": " + e.getMessage(), e);
                }
            }
        }
    }
}

import com.google.gson.Gson;

/**
 * Loader for block rules configuration from JSON.
 *
 * Reads {@code /data/weylandyutani/block_rules.json}.
 *
 * Configuration flow:
 *
 * <pre>
 * JSON Resource
 *      ↓
 * ResourceLoadingHelper
 *      ↓
 * BlockRulesConfig
 *      ↓
 * BlockRulesStorage
 *      ↓
 * Runtime Systems
 * </pre>
 *
 * This loader does not provide fallback gameplay defaults.
 * Missing or invalid configuration is considered a configuration contract failure.
 *
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public final class BlockRulesConfigLoader implements ConfigLoader {

    private static final String RESOURCE_PATH =
            "/data/weylandyutani/block_rules.json";

    private static final String LOADER_NAME =
            "BlockRulesConfigLoader";

    private static final BlockRulesConfigLoader INSTANCE =
            new BlockRulesConfigLoader();


    private BlockRulesConfigLoader() {
    }


    /**
     * Get singleton loader instance.
     *
     * @return loader instance
     */
    public static BlockRulesConfigLoader getInstance() {
        return INSTANCE;
    }


    /**
     * Load block rules configuration from JSON resource.
     *
     * @throws IllegalStateException if configuration cannot be loaded
     */
    @Override
    public void loadConfigs() {

        try {
            JsonObject json =
                    ResourceLoadingHelper.loadJsonFromResource(RESOURCE_PATH);


            if (json == null) {
                throw new IllegalStateException(
                        "Missing required configuration resource: "
                                + RESOURCE_PATH);
            }


            BlockRulesConfig config =
                    new Gson().fromJson(json, BlockRulesConfig.class);


            if (config == null) {
                throw new IllegalStateException(
                        "Failed to deserialize block rules configuration");
            }


            /*
             * TODO:
             * Replace this with BlockRulesConfigValidator once created.
             *
             * Expected validation:
             * - required collections exist
             * - block IDs are not blank
             * - no duplicate entries
             * - no conflicting breakable/unbreakable entries
             */
            validateConfig(config);


            BlockRulesStorage.setConfig(config);


            ModLogger.debug(
                    LOADER_NAME,
                    String.format(
                            "Loaded %d unbreakable blocks, %d breakable blocks",
                            config.unbreakable.size(),
                            config.breakable.size()));

        } catch (JsonSyntaxException e) {

            ModLogger.error(
                    LOADER_NAME,
                    "Invalid JSON syntax in block_rules.json: "
                            + e.getMessage(),
                    e);

            throw new IllegalStateException(
                    "Invalid block rules configuration",
                    e);

        } catch (Exception e) {

            ModLogger.error(
                    LOADER_NAME,
                    "Failed loading block rules configuration: "
                            + e.getMessage(),
                    e);

            throw new IllegalStateException(
                    "Unable to load block rules configuration",
                    e);
        }
    }


    /**
     * Clears existing configuration and reloads from resource.
     */
    @Override
    public void reloadConfigs() {
        BlockRulesStorage.clear();
        loadConfigs();
    }


    /**
     * Check whether configuration has been loaded.
     *
     * @return true when configuration exists
     */
    @Override
    public boolean isLoaded() {
        return BlockRulesStorage.isLoaded();
    }


    /**
     * Returns number of configured block rules.
     *
     * @return total configured rules
     */
    @Override
    public int getConfigCount() {

        BlockRulesConfig config =
                BlockRulesStorage.getConfig();

        if (config == null) {
            return 0;
        }

        return config.unbreakable.size()
                + config.breakable.size();
    }


    /**
     * Returns loader identifier.
     *
     * @return loader name
     */
    @Override
    public String getLoaderName() {
        return LOADER_NAME;
    }


    /**
     * Confirms configuration is loaded.
     *
     * @return true if loaded
     */
    @Override
    public boolean validateAllConfigs() {
        return BlockRulesStorage.isLoaded();
    }


    /**
     * Temporary validation hook.
     *
     * This should eventually move into:
     *
     * layer_1_infrastructure.level_X.block.BlockRulesConfigValidator
     *
     * @param config loaded configuration
     */
    private static void validateConfig(BlockRulesConfig config) {

        if (config.unbreakable == null) {
            throw new IllegalStateException(
                    "Block rules missing unbreakable list");
        }

        if (config.breakable == null) {
            throw new IllegalStateException(
                    "Block rules missing breakable list");
        }
    }
}

        EntityConfig.SpeciesVariantConfig config = new EntityConfig.SpeciesVariantConfig();

        config.variantId = JsonParsingHelper.getString(jsonData, "variant_id", "");
        config.hostType = JsonParsingHelper.getString(jsonData, "host_type", "");

        config.speedMultiplier = JsonParsingHelper.getFloat(jsonData, "speed_multiplier", 1.0f);
        config.damageMultiplier = JsonParsingHelper.getFloat(jsonData, "damage_multiplier", 1.0f);
        config.intelligenceBonus = JsonParsingHelper.getFloat(jsonData, "intelligence_bonus", 0.0f);

        config.wallClimbing = readSpeciesBooleanAlias(jsonData, "climbing_ability", "wall_climbing", false);
        config.webCreation = readSpeciesBooleanAlias(jsonData, "web_abilities", "web_creation", false);
        config.explosiveAbility = readSpeciesBooleanAlias(jsonData, "infection_spread", "explosive_ability", false);
        config.teleportation = JsonParsingHelper.getBoolean(jsonData, "teleportation", false);
        config.dimensionalAwareness = JsonParsingHelper.getBoolean(jsonData, "dimensional_awareness", false);

        config.aggressionBonus = JsonParsingHelper.getFloat(jsonData, "aggression_bonus", 1.0f);
        config.stealthBonus = JsonParsingHelper.getFloat(jsonData, "stealth_bonus", 1.0f);
        config.nocturnalPreference = JsonParsingHelper.getBoolean(jsonData, "nocturnal_preference", false);
        config.overallScale = JsonParsingHelper.getFloat(jsonData, "overall_scale", 1.0f);
        config.muscleDefinition = JsonParsingHelper.getFloat(jsonData, "muscle_definition", 1.0f);

        config.caveBiomes = JsonParsingHelper.getBoolean(jsonData, "cave_biomes", false);
        config.endBiomes = JsonParsingHelper.getBoolean(jsonData, "end_biomes", false);
        config.surfaceBiomes = JsonParsingHelper.getBoolean(jsonData, "surface_biomes", false);
        config.territorySize = JsonParsingHelper.getString(jsonData, "territory_size", "medium");

        return config;
    }

    private static boolean readSpeciesBooleanAlias(
            @Nonnull JsonObject jsonData,
            @Nonnull String canonicalKey,
            @Nonnull String aliasKey,
            boolean fallback) {
        if (jsonData.has(canonicalKey)) {
            return jsonData.get(canonicalKey).getAsBoolean();
        }
        if (jsonData.has(aliasKey)) {
            return jsonData.get(aliasKey).getAsBoolean();
        }
        return fallback;
    }

    /**
     * Loads and parses a JSON file into a JsonObject.
     *
     * @param resourcePath The resource path to the JSON file
     * @return The parsed JsonObject, or null if file not found
     */
    public static JsonObject loadJsonFromResource(String resourcePath) {
        return ResourceLoadingHelper.loadJsonFromResource(resourcePath, EntityConfigParser.class);
    }
}

package com.weylandyutani.layer_1_infrastructure.level_2.registry;

import com.google.gson.JsonObject;
import com.weylandyutani.layer_1_infrastructure.level_0.io.ResourceLoader;
import com.weylandyutani.layer_1_infrastructure.level_0.platform.ModLogger;
import com.weylandyutani.layer_1_infrastructure.level_1.io.ClasspathResourceLoader;
import com.weylandyutani.layer_1_infrastructure.level_1.registry.ConfigLoaderBase;

/**
 * Base class for static config loaders that use classpath resource loading.
 * Provides common loading/storage patterns for loaders that don't use ResourceManager.
 * 
 * FILE STATUS: STABLE
 * 
 * This class implements DRY by centralizing the common pattern used by:
 * - EffectConfigLoader
 * - HostConfigLoader
 * - MobEntityConfigLoader
 * - BlockConfigLoader
 * 
 * Uses ClasspathResourceLoader by default, but can be configured with different ResourceLoader.
 * 
 * @param <T> The type of configuration object this loader manages
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public abstract class StaticConfigLoader<T> extends ConfigLoaderBase<T> {
    
    /**
     * The resource loader strategy (defaults to ClasspathResourceLoader).
     */
    protected ResourceLoader resourceLoader;
    
    /**
     * The context class for resource loading (usually the loader class itself).
     */
    protected final Class<?> contextClass;
    
    /**
     * Constructs a StaticConfigLoader with the specified name and context class.
     * 
     * @param loaderName The name of this loader for logging purposes
     * @param contextClass The class to use for resource loading (usually the loader class)
     */
    protected StaticConfigLoader(String loaderName, Class<?> contextClass) {
        super(loaderName);
        this.contextClass = contextClass;
        this.resourceLoader = ClasspathResourceLoader.INSTANCE;
    }
    
    /**
     * Sets a custom resource loader strategy.
     * 
     * @param resourceLoader The resource loader to use
     */
    protected void setResourceLoader(ResourceLoader resourceLoader) {
        this.resourceLoader = resourceLoader;
    }
    
    /**
     * Gets the resource path for a specific configuration.
     * Subclasses must implement this to define where their configs are located.
     * 
     * @param configName The name of the configuration
     * @return The resource path for the configuration
     */
    protected abstract String getResourcePath(String configName);
    
    /**
     * Parses JSON data into a configuration object.
     * Subclasses must implement this to define their parsing logic.
     * 
     * @param jsonData The JSON object to parse
     * @return The parsed configuration object
     */
    protected abstract T parseConfig(JsonObject jsonData);
    
    /**
     * Gets the default configuration names to load.
     * Subclasses must implement this to define which configs to load by default.
     * 
     * @return Array of default configuration names
     */
    protected abstract String[] getDefaultConfigNames();
    
    @Override
    public void loadConfigs() {
        if (loaded) {
            ModLogger.debug(loaderName + " configurations already loaded");
            return;
        }
        
        ModLogger.loading("Loading " + loaderName + " configurations...");
        ModLogger.startTimer(loaderName + "_load");
        
        try {
            clearConfigs();
            String[] configNames = getDefaultConfigNames();
            
            for (String configName : configNames) {
                loadConfig(configName);
            }
            
            setLoaded(true);
            ModLogger.endTimer(loaderName + "_load");
            ModLogger.success("Loaded " + getConfigCount() + " " + loaderName + " configurations");
            
        } catch (Exception e) {
            ModLogger.error("Failed to load " + loaderName + " configurations", e);
            throw new RuntimeException("Failed to load " + loaderName + " configurations", e);
        }
    }
    
    /**
     * Loads a specific configuration.
     * 
     * @param configName The name of the configuration to load
     */
    protected void loadConfig(String configName) {
        try {
            String resourcePath = getResourcePath(configName);
            JsonObject jsonData = resourceLoader.loadJson(resourcePath, contextClass);
            
            if (jsonData == null) {
                ModLogger.warn(loaderName + " config not found: " + configName);
                return;
            }
            
            T config = parseConfig(jsonData);
            
            if (validateConfig(config, configName)) {
                putConfig(configName, config);
                ModLogger.success("Loaded " + loaderName + " config: " + configName);
            } else {
                ModLogger.warn("Invalid " + loaderName + " config: " + configName);
            }
        } catch (Exception e) {
            ModLogger.error("Failed to load " + loaderName + " config " + configName, e);
            throw new RuntimeException("Failed to load " + loaderName + " config " + configName, e);
        }
    }
    
    @Override
    public void reloadConfigs() {
        ModLogger.loading("Reloading " + loaderName + " configurations...");
        clearConfigs();
        setLoaded(false);
        loadConfigs();
    }
    
    
    /**
     * Gets all loaded configuration names.
     * 
     * @return Set of all loaded configuration names
     */
    public java.util.Set<String> getLoadedConfigNames() {
        return getConfigNames();
    }
}

/**
 * Orchestrator for block configuration loading (textures then block visuals).
 *
 * FILE STATUS: FUNCTIONAL
 */
public class BlockConfigLoader extends StaticConfigLoader<LoadedBlockVisualConfig> {

    private static final BlockConfigLoader INSTANCE = new BlockConfigLoader();

    private BlockConfigLoader() {
        super("BlockConfigLoader", BlockConfigLoader.class);
    }

    public static BlockConfigLoader getInstance() {
        return INSTANCE;
    }

    public static void loadBlockTextures() {
        if (BlockTextureRegistry.isLoaded()) {
            ModLogger.debug("BlockConfigLoader", "Block textures already loaded");
            return;
        }

        ModLogger.info("BlockConfigLoader", "Loading block texture registry...");

        try {
            String resourcePath = "/data/weylandyutani/block_textures.json";
            JsonObject jsonData = INSTANCE.resourceLoader.loadJson(resourcePath, INSTANCE.contextClass);

            if (jsonData == null) {
                throw new ConfigContractException(
                        "BlockConfigLoader", resourcePath, "block textures file not found");
            }

            for (String key : jsonData.keySet()) {
                if (key.startsWith("_")) {
                    continue;
                }

                if (jsonData.get(key).isJsonObject()) {
                    JsonObject textureData = Objects.requireNonNull(jsonData.getAsJsonObject(key));
                    LoadedBlockTextureConfig config = BlockConfigParser.parseBlockTextureConfig(textureData);
                    BlockTextureRegistry.put(key, config);
                    ModLogger.info("BlockConfigLoader",
                            String.format("Loaded block texture config: %s (type: %s)", key, config.type));
                }
            }

            BlockTextureRegistry.setLoaded(true);
            ModLogger.info("BlockConfigLoader",
                    String.format("Loaded %d block texture configurations", BlockTextureRegistry.size()));

        } catch (ConfigContractException e) {
            throw e;
        } catch (Exception e) {
            ModLogger.error("BlockConfigLoader", "Failed to load block textures file", e);
            throw new ConfigContractException(
                    "BlockConfigLoader", "/data/weylandyutani/block_textures.json",
                    "load failed: " + e.getMessage());
        }
    }

    @Override
    public void loadConfigs() {
        if (loaded) {
            ModLogger.debug("BlockConfigLoader", "Block configurations already loaded");
            return;
        }

        if (!BlockTextureRegistry.isLoaded()) {
            loadBlockTextures();
        }

        ModLogger.info("BlockConfigLoader", "Loading block configurations...");

        loadBlockConfig("resin_block");
        loadBlockConfig("spiny_block");
        loadBlockConfig("cocoon_block");
        loadBlockConfig("lab_machine");
        loadBlockConfig("specimen_glass");
        loadBlockConfig("specimen_pod_controller");

        setLoaded(true);
        ModLogger.info("BlockConfigLoader",
                String.format("Loaded %d block configurations", getConfigCount()));
    }

    private void loadBlockConfig(String blockName) {
        try {
            String resourcePath = "/data/" + ModContext.modId() + "/blocks/" + blockName + ".json";
            JsonObject jsonData = resourceLoader.loadJson(resourcePath, contextClass);

            if (jsonData == null) {
                throw new ConfigContractException(
                        "BlockConfigLoader", resourcePath, "block config not found");
            }

            LoadedBlockVisualConfig config = BlockConfigParser.parseBlockConfig(jsonData);
            putConfig(blockName, config);
            ModLogger.info("BlockConfigLoader",
                    String.format("Loaded block config: %s (texture_key: %s)", blockName, config.textureKey));

        } catch (ConfigContractException e) {
            throw e;
        } catch (Exception e) {
            ModLogger.error("BlockConfigLoader",
                    String.format("Failed to load block config: %s", blockName), e);
            throw new ConfigContractException(
                    "BlockConfigLoader", "/data/weylandyutani/blocks/" + blockName + ".json",
                    "load failed: " + e.getMessage());
        }
    }

    public static void loadBlockConfigs() {
        INSTANCE.loadConfigs();
    }

    public static LoadedBlockVisualConfig getConfig(String blockName) {
        if (!INSTANCE.isLoaded()) {
            ModLogger.warn("BlockConfigLoader",
                    String.format("Block configs not loaded yet, attempting to load: %s", blockName));
            loadBlockConfigs();
        }
        return INSTANCE.configs.get(blockName);
    }

    public static boolean isBlockConfigLoaded() {
        return INSTANCE.loaded;
    }

    public static LoadedBlockTextureConfig getBlockTextureConfig(String textureKey) {
        if (!BlockTextureRegistry.isLoaded()) {
            ModLogger.warn("BlockConfigLoader", "Block textures not loaded yet, attempting to load");
            loadBlockTextures();
        }
        return BlockTextureRegistry.get(textureKey);
    }

    public static String getSingleTexture(String blockName) {
        return BlockTextureResolver.resolveSingle(getConfig(blockName));
    }

    public static String getMultiTexture(String blockName, String side) {
        return BlockTextureResolver.resolveMulti(getConfig(blockName), side);
    }

    public static void reloadBlockConfigs() {
        ModLogger.info("BlockConfigLoader", "Reloading block configurations...");
        INSTANCE.clearConfigs();
        BlockTextureRegistry.clear();
        INSTANCE.setLoaded(false);
        loadBlockTextures();
        loadBlockConfigs();
    }

    @Override
    public void reloadConfigs() {
        reloadBlockConfigs();
    }

    @Override
    protected String getResourcePath(String configName) {
        return "";
    }

    @Override
    protected LoadedBlockVisualConfig parseConfig(JsonObject jsonData) {
        return BlockConfigParser.parseBlockConfig(Objects.requireNonNull(jsonData));
    }

    @Override
    protected String[] getDefaultConfigNames() {
        return new String[]{};
    }
}

/**
 * Orchestrator for entity configuration loading.
 * Shipped JSON must satisfy contracts; load failures abort mod initialization.
 *
 * <p>Mob-based xenomorph variants are owned solely by the lifecycle mob-variant loader
 * (see {@code MobVariantManifest}) — this loader does not re-parse those resources.
 */
public class EntityConfigLoader {

    public static void loadConfigurations() {
        ModLogger.info("EntityConfigLoader", "Starting entity configuration loading...");

        if (EntityConfigStorage.isLoaded()) {
            ModLogger.info("EntityConfigLoader", "Entity configs already loaded, skipping");
            return;
        }

        EntityConfigStorage.clear();

        for (XenomorphType type : XenomorphType.values()) {
            String resourcePath = getResourcePath(type);
            ModLogger.info("EntityConfigLoader", "Loading config for: " + type.getConfigKey());

            JsonObject jsonData = EntityConfigParser.loadJsonFromResource(resourcePath);
            if (jsonData == null) {
                throw new ConfigContractException("EntityConfigLoader", resourcePath, "config file not found");
            }

            EntityConfig config;
            try {
                config = EntityConfigParser.parseEntityConfig(jsonData);
            } catch (Exception e) {
                throw new ConfigContractException("EntityConfigLoader", resourcePath,
                        "parse failed: " + e.getMessage());
            }

            EntityConfigValidator.validateConfig(config, type, resourcePath);
            EntityConfigStorage.storeConfig(type, config);
            ModLogger.info("EntityConfigLoader", "Loaded entity config for: " + type.getConfigKey());
        }

        EntityConfigStorage.setLoaded(true);
        ModLogger.info("EntityConfigLoader",
                "Entity configurations loaded successfully - " + EntityConfigStorage.getConfigCount() + " entities");
    }

    private static String getResourcePath(XenomorphType type) {
        return EntityConfigPaths.resourcePath(type);
    }
}

            throw new ConfigContractException("LifecycleConfig", getResourcePath("lifecycle"),
                    "enabled", "required");
        }
        JsonObject defaults = JsonParsingHelper.getJsonObject(jsonData, "defaults");
        JsonObject attachment = JsonParsingHelper.getJsonObject(jsonData, "attachment");
        int attachmentDuration = defaults != null
                ? JsonParsingHelper.getInt(defaults, "attachment_duration", 0) : 0;
        int implantationDelay = defaults != null

        int spentWanderTicks = defaults != null

        int effectRefreshTicks = attachment != null
                ? JsonParsingHelper.getInt(attachment, "effect_refresh_ticks", 0) : 0;
        if (attachmentDuration <= 0 || implantationDelay <= 0 || spentWanderTicks <= 0
                || deathAnimationTicks <= 0 || effectRefreshTicks <= 0) {
            throw new ConfigContractException("LifecycleConfig", getResourcePath("lifecycle"),
                    "defaults/attachment", "required positive tuning values");
        }
        FormThresholds formThresholds = parseFormThresholds(jsonData);
        MaturationPolicy maturationPolicy = parseMaturationPolicy(jsonData);
        return new Settings(
                jsonData.get("enabled").getAsBoolean(),
                attachmentDuration,
                implantationDelay,
                spentWanderTicks,
                deathAnimationTicks,
                effectRefreshTicks,
                formThresholds,
                maturationPolicy);
    }

    private static MaturationPolicy parseMaturationPolicy(@Nonnull JsonObject jsonData) {
        JsonObject policy = JsonParsingHelper.getJsonObject(jsonData, "maturation_policy");
        if (policy == null) {
            throw new ConfigContractException("LifecycleConfig", LIFECYCLE_RESOURCE_PATH,
                    "maturation_policy", "required");
        }
        double foodHuntUrgencyThreshold = JsonParsingHelper.getDouble(policy, "food_hunt_urgency_threshold");
        int defaultFoodTimerReductionTicks = JsonParsingHelper.getInt(policy, "default_food_timer_reduction_ticks");
        if (foodHuntUrgencyThreshold <= 0.0 || foodHuntUrgencyThreshold > 1.0) {
            throw new ConfigContractException("LifecycleConfig", LIFECYCLE_RESOURCE_PATH,
                    "maturation_policy.food_hunt_urgency_threshold", "must be in (0, 1]");
        }
        if (defaultFoodTimerReductionTicks <= 0) {
            throw new ConfigContractException("LifecycleConfig", LIFECYCLE_RESOURCE_PATH,
                    "maturation_policy.default_food_timer_reduction_ticks", "must be > 0");
        }
        return new MaturationPolicy(foodHuntUrgencyThreshold, defaultFoodTimerReductionTicks);
    }

    private static FormThresholds parseFormThresholds(@Nonnull JsonObject jsonData) {
        JsonObject thresholds = JsonParsingHelper.getJsonObject(jsonData, "form_thresholds");
        if (thresholds == null) {

                    "form_thresholds", "required");

        if (catalogKey == null || catalogKey.isBlank()) {
            return HostType.LIVING_BIOLOGICAL;
        }
        for (HostType type : HostType.values()) {
            if (type.getKey().equalsIgnoreCase(catalogKey) || type.name().equalsIgnoreCase(catalogKey)) {
                return type;
            }
        }
        return HostType.LIVING_BIOLOGICAL;

            } catch (Exception e) {
                throw new ConfigContractException(loaderName, path, "load failed: " + e.getMessage());
            }
        }
        loaded = true;
        ModLogger.info(loaderName, "Loaded " + configs.size() + " mob variant configs");
    }

package com.weylandyutani.layer_1_infrastructure.level_4.entity;

import com.weylandyutani.layer_1_infrastructure.level_0.effects.EffectConfig;
import com.weylandyutani.layer_1_infrastructure.level_0.platform.ModLogger;
import com.weylandyutani.layer_1_infrastructure.level_3.registry.EffectConfigLoader;
import java.util.function.Function;

/**
 * Enhanced utility class for accessing configuration values with consistent error handling.
 * 
 * This class centralizes repetitive config access patterns and provides
 * standardized error handling for configuration loading failures.
 * 
 * Features:
 * - Null-safety checks for all parameters
 * - Performance logging for expensive operations
 * - Detailed error messages with context
 * - Validation helpers for common data types
 * 
 * <h2>Usage Examples</h2>
 * <pre>{@code
 * // Get a simple config value
 * int duration = ConfigAccessHelper.getConfigValue(
 *     "gestation",
 *     config -> config.duration.durationTicks,
 *     "Invalid gestation duration"
 * );
 * 
 * // Get a nested config value with validation
 * int warningStart = ConfigAccessHelper.getConfigValue(
 *     "gestation",
 *     config -> config.visualEffects.warningStartTicks,
 *     "Invalid warning phase start"
 * );
 * }</pre>
 * 
 * FILE STATUS: STABLE
 * 
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class ConfigAccessHelper {
    
    /**
     * Gets a configuration value from the specified effect configuration.
     * 
     * This method handles the common pattern of:
     * 1. Loading the effect configuration
     * 2. Extracting the desired value using the provided function
     * 3. Validating the result
     * 4. Throwing descriptive errors if any step fails
     * 
     * @param <T> The type of the configuration value to extract
     * @param effectName The name of the effect configuration to load
     * @param extractor Function to extract the desired value from the config
     * @param errorMessage Error message to use if extraction fails
     * @return The extracted configuration value
     * @throws RuntimeException If configuration cannot be loaded or value is invalid
     */
    public static <T> T getConfigValue(
            String effectName,
            Function<EffectConfig, T> extractor,
            String errorMessage) {
        
        // Validate input parameters
        if (effectName == null || effectName.trim().isEmpty()) {
            ModLogger.error("Cannot get config value: effectName is null or empty");
            throw new IllegalArgumentException("Effect name cannot be null or empty");
        }
        
        if (extractor == null) {
            ModLogger.error("Cannot get config value for '" + effectName + "': extractor is null");
            throw new IllegalArgumentException("Extractor function cannot be null");
        }
        
        if (errorMessage == null || errorMessage.trim().isEmpty()) {
            ModLogger.error("Cannot get config value for '" + effectName + "': errorMessage is null or empty");
            throw new IllegalArgumentException("Error message cannot be null or empty");
        }
        
        // Start performance timing
        ModLogger.startTimer("ConfigAccess_" + effectName);
        
        try {
            // Load the configuration
            var config = EffectConfigLoader.getConfig(effectName);
            if (config == null) {
                ModLogger.error("Configuration not found: " + effectName);
                throw new RuntimeException(effectName + " configuration not found - check JSON config file");
            }
            
            // Extract the value
            T value;
            try {
                value = extractor.apply(config);
            } catch (Exception e) {
                ModLogger.error("Failed to extract " + errorMessage + " from " + effectName + " config: " + e.getMessage(), e);
                throw new RuntimeException("Failed to extract " + errorMessage + " from " + effectName + " config: " + e.getMessage());
            }
            
            // Validate the value
            if (value == null) {
                ModLogger.error(errorMessage + " is null in " + effectName + " configuration");
                throw new RuntimeException(errorMessage + " is null in " + effectName + " configuration");
            }
            
            ModLogger.debug("Successfully retrieved config value for: " + effectName);
            return value;
        } finally {
            ModLogger.endTimer("ConfigAccess_" + effectName);
        }
    }
    
    /**
     * Gets a configuration value with custom validation.
     * 
     * This method extends the basic getConfigValue method by allowing
     * custom validation logic to be applied to the extracted value.
     * 
     * @param <T> The type of the configuration value to extract
     * @param effectName The name of the effect configuration to load
     * @param extractor Function to extract the desired value from the config
     * @param validator Function to validate the extracted value
     * @param errorMessage Error message to use if validation fails
     * @return The extracted and validated configuration value
     * @throws RuntimeException If configuration cannot be loaded or validation fails
     */
    public static <T> T getConfigValueWithValidation(
            String effectName,
            Function<EffectConfig, T> extractor,
            Function<T, Boolean> validator,
            String errorMessage) {
        
        T value = getConfigValue(effectName, extractor, errorMessage);
        
        if (!validator.apply(value)) {
            throw new RuntimeException("Invalid " + errorMessage + " in " + effectName + " config: " + value);
        }
        
        return value;
    }
    
    /**
     * Gets an integer configuration value with range validation.
     * 
     * Convenience method for integer values that need to be within a specific range.
     * 
     * @param effectName The name of the effect configuration to load
     * @param extractor Function to extract the integer value from the config
     * @param minValue Minimum allowed value (inclusive)
     * @param maxValue Maximum allowed value (inclusive)
     * @param errorMessage Error message to use if validation fails
     * @return The validated integer value
     * @throws RuntimeException If configuration cannot be loaded or value is out of range
     */
    public static int getIntConfigValue(
            String effectName,
            Function<EffectConfig, Integer> extractor,
            int minValue,
            int maxValue,
            String errorMessage) {
        
        return getConfigValueWithValidation(
            effectName,
            extractor,
            value -> value >= minValue && value <= maxValue,
            errorMessage + " must be between " + minValue + " and " + maxValue
        );
    }
    
    /**
     * Gets a float configuration value with range validation.
     * 
     * Convenience method for float values that need to be within a specific range.
     * 
     * @param effectName The name of the effect configuration to load
     * @param extractor Function to extract the float value from the config
     * @param minValue Minimum allowed value (inclusive)
     * @param maxValue Maximum allowed value (inclusive)
     * @param errorMessage Error message to use if validation fails
     * @return The validated float value
     * @throws RuntimeException If configuration cannot be loaded or value is out of range
     */
    public static float getFloatConfigValue(
            String effectName,
            Function<EffectConfig, Float> extractor,
            float minValue,
            float maxValue,
            String errorMessage) {
        
        return getConfigValueWithValidation(
            effectName,
            extractor,
            value -> value >= minValue && value <= maxValue,
            errorMessage + " must be between " + minValue + " and " + maxValue
        );
    }
    
    // Private constructor to prevent instantiation
    private ConfigAccessHelper() {
        throw new UnsupportedOperationException("ConfigAccessHelper is a utility class and cannot be instantiated");
    }
}

/**

 * Allows EntityConfigLoader to be managed by ConfigLoaderRegistry.
 * 
 * FILE STATUS: STABLE
 * 
 * This adapter implements the Adapter pattern, allowing EntityConfigLoader
 * (which uses a different orchestration pattern) to integrate with the unified registry.
 * 
 * @author WeylandYutaniCorp
 * @since 0.1.0
 */
public class EntityConfigLoaderAdapter implements ConfigLoader {
    
    /**
     * Singleton instance for static access.
     */
    private static final EntityConfigLoaderAdapter INSTANCE = new EntityConfigLoaderAdapter();
    
    /**
     * Private constructor for singleton pattern.
     */
    private EntityConfigLoaderAdapter() {
        // Singleton
    }
    
    /**
     * Gets the singleton instance.
     * 
     * @return The singleton instance
     */
    public static EntityConfigLoaderAdapter getInstance() {
        return INSTANCE;
    }
    
    @Override
    public void loadConfigs() {
        EntityConfigLoader.loadConfigurations();
    }
    
    @Override
    public void reloadConfigs() {
        EntityConfigStorage.clear();
        EntityConfigStorage.setLoaded(false);
        EntityConfigLoader.loadConfigurations();
    }
    
    @Override
    public boolean isLoaded() {
        return EntityConfigStorage.isLoaded();
    }
    
    @Override
    public int getConfigCount() {
        return EntityConfigStorage.getConfigCount();
    }
    
    @Override
    public String getLoaderName() {
        return "EntityConfigLoader";
    }
    
    @Override
    public boolean validateAllConfigs() {
        // EntityConfigLoader validates during loading, so all stored configs are valid
        // But we can double-check by verifying storage is loaded
        if (!EntityConfigStorage.isLoaded()) {
            return false;
        }
        
        // All configs in storage were validated during loading
        return EntityConfigStorage.getConfigCount() > 0;
    }
}
